22#include "test.h"
23
24#include "memdebug.h"
25
26int test(char *URL)
27{
28 const unsigned char a[] = {0x2f, 0x3a, 0x3b, 0x3c, 0x3d, 0x3e, 0x3f,
29 0x91, 0xa2, 0xb3, 0xc4, 0xd5, 0xe6, 0xf7};
30 CURLcode res = CURLE_OK;
31 char *ptr = NULL;
32 int asize;
33 int outlen;
34 char *raw;
35
36 (void)URL; /* we don't use this */
37
38 if(curl_global_init(CURL_GLOBAL_ALL) != CURLE_OK) {
39 fprintf(stderr, "curl_global_init() failed\n");
40 return TEST_ERR_MAJOR_BAD;
41 }
42
43 asize = (int)sizeof(a);
44 ptr = curl_easy_escape(NULL, (char *)a, asize);
45 printf("%s\n", ptr);
46 if(ptr)
47 curl_free(ptr);
48
49 /* deprecated API */
50 ptr = curl_escape((char *)a, asize);
51 printf("%s\n", ptr);
52 if(!ptr) {
53 res = TEST_ERR_MAJOR_BAD;
54 goto test_cleanup;
55 }
56
57 raw = curl_easy_unescape(NULL, ptr, (int)strlen(ptr), &outlen);
58 printf("outlen == %d\n", outlen);
59 printf("unescape == original? %s\n",
60 memcmp(raw, a, outlen) ? "no" : "YES");
61 if(raw)
62 curl_free(raw);
63
64 /* deprecated API */
65 raw = curl_unescape(ptr, (int)strlen(ptr));
66 if(!raw) {
67 res = TEST_ERR_MAJOR_BAD;
68 goto test_cleanup;
69 }
70 outlen = (int)strlen(raw);
71 printf("[old] outlen == %d\n", outlen);
72 printf("[old] unescape == original? %s\n",
73 memcmp(raw, a, outlen) ? "no" : "YES");
74 if(raw)
75 curl_free(raw);
76 if(ptr)
77 curl_free(ptr);
78
79 /* weird input length */
80 ptr = curl_easy_escape(NULL, (char *)a, -1);
81 printf("escape -1 length: %s\n", ptr);
82
83 /* weird input length */
84 outlen = 2017; /* just a value */
85 ptr = curl_easy_unescape(NULL, (char *)"moahahaha", -1, &outlen);
86 printf("unescape -1 length: %s %d\n", ptr, outlen);
87
88test_cleanup:
89 if(ptr)
90 curl_free(ptr);
91 curl_global_cleanup();
92
93 return (int)res;
94}
