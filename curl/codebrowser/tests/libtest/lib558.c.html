26int test(char *URL)
27{
28 unsigned char a[] = {0x2f, 0x3a, 0x3b, 0x3c, 0x3d, 0x3e, 0x3f,
29 0x91, 0xa2, 0xb3, 0xc4, 0xd5, 0xe6, 0xf7};
30 CURLcode res = CURLE_OK;
31 char *ptr = NULL;
32 int asize;
33
34 (void)URL; /* we don't use this */
35
36 if(curl_global_init(CURL_GLOBAL_ALL) != CURLE_OK) {
37 fprintf(stderr, "curl_global_init() failed\n");
38 return TEST_ERR_MAJOR_BAD;
39 }
40
41 ptr = malloc(558);
42 Curl_safefree(ptr);
43
44 asize = (int)sizeof(a);
45 ptr = curl_easy_escape(NULL, (char *)a, asize);
46 if(ptr)
47 curl_free(ptr);
48
49 curl_global_cleanup();
50
51 return (int)res;
52}
