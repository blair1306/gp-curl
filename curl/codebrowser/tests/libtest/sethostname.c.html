24#include "sethostname.h"

26/*
27 * we force our own host name, in order to make some tests machine independent
28 */
29
30int gethostname(char *name, GETHOSTNAME_TYPE_ARG2 namelen)
31{
32 const char *force_hostname = getenv("CURL_GETHOSTNAME");
33 if(force_hostname) {
34 strncpy(name, force_hostname, namelen);
35 name[namelen-1] = '\0';
36 return 0;
37 }
38
39 /* LD_PRELOAD used, but no hostname set, we'll just return a failure */
40 return -1;
41}
