22#include "test.h"
23
24#include "memdebug.h"
25
26#include <curl/multi.h>
27
28int test(char *URL)
29{
30 CURLM *handle;
31 int res = CURLE_OK;
32 static const char * const bl_servers[] =
33 {"Microsoft-IIS/6.0", "nginx/0.8.54", NULL};
34 static const char * const bl_sites[] =
35 {"curl.haxx.se:443", "example.com:80", NULL};
36
37 global_init(CURL_GLOBAL_ALL);
38 handle = curl_multi_init();
39 (void)URL; /* unused */
40
41 curl_multi_setopt(handle, CURLMOPT_PIPELINING_SERVER_BL, bl_servers);
42 curl_multi_setopt(handle, CURLMOPT_PIPELINING_SITE_BL, bl_sites);
43 curl_multi_cleanup(handle);
44 curl_global_cleanup();
45 return 0;
46}
