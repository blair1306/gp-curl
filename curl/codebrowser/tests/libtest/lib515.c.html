22#include "test.h"
23
24#include "memdebug.h"
25
26int test(char *URL)
27{
28 CURL *curl;
29 CURLcode res = CURLE_OK;
30
31 if(curl_global_init(CURL_GLOBAL_ALL) != CURLE_OK) {
32 fprintf(stderr, "curl_global_init() failed\n");
33 return TEST_ERR_MAJOR_BAD;
34 }
35
36 curl = curl_easy_init();
37 if(!curl) {
38 fprintf(stderr, "curl_easy_init() failed\n");
39 curl_global_cleanup();
40 return TEST_ERR_MAJOR_BAD;
41 }
42
43 /* First set the URL that is about to receive our POST. */
44 test_setopt(curl, CURLOPT_URL, URL);
45 test_setopt(curl, CURLOPT_POSTFIELDS, NULL);
46 test_setopt(curl, CURLOPT_POSTFIELDSIZE, 0L);
47 test_setopt(curl, CURLOPT_VERBOSE, 1L); /* show verbose for debug */
48 test_setopt(curl, CURLOPT_HEADER, 1L); /* include header */
49
50 /* Now, we should be making a zero byte POST request */
51 res = curl_easy_perform(curl);
52
53test_cleanup:
54
55 /* always cleanup */
56 curl_easy_cleanup(curl);
57 curl_global_cleanup();
58
59 return (int)res;
60}
