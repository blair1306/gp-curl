22#include "curl_setup.h"
23
24#include "curl_gethostname.h"
25
26#define HOSTNAME_MAX 1024
27
28int main(int argc, char *argv[])
29{
30 char buff[HOSTNAME_MAX];
31 if(argc != 2) {
32 printf("Usage: %s EXPECTED_HOSTNAME\n", argv[0]);
33 return 1;
34 }
35
36 if(Curl_gethostname(buff, HOSTNAME_MAX)) {
37 printf("Curl_gethostname() failed\n");
38 return 1;
39 }
40
41 /* compare the name returned by Curl_gethostname() with the expected one */
42 if(strncmp(buff, argv[1], HOSTNAME_MAX)) {
43 printf("got unexpected host name back, LD_PRELOAD failed\n");
44 return 1;
45 }
46 return 0;
47}
