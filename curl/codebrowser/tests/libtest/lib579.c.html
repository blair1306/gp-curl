26static const char * const post[]={
27 "one",
28 "two",
29 "three",
30 "and a final longer crap: four",
31 NULL
32};

34
35struct WriteThis {
36 int counter;
37};
38
39static int progress_callback(void *clientp, double dltotal, double dlnow,
40 double ultotal, double ulnow)
41{
42 FILE *moo;
43 static int prev_ultotal = -1;
44 static int prev_ulnow = -1;
45 (void)clientp; /* UNUSED */
46 (void)dltotal; /* UNUSED */
47 (void)dlnow; /* UNUSED */
48
49 /* to avoid depending on timing, which will cause this progress function to
50 get called a different number of times depending on circumstances, we
51 only log these lines if the numbers are different from the previous
52 invoke */
53 if((prev_ultotal != (int)ultotal) ||
54 (prev_ulnow != (int)ulnow)) {
55
56 moo = fopen(libtest_arg2, "ab");
57 if(moo) {
58 fprintf(moo, "Progress callback called with UL %d out of %d\n",
59 (int)ulnow, (int)ultotal);
60 fclose(moo);
61 }
62 prev_ulnow = (int) ulnow;
63 prev_ultotal = (int) ultotal;
64 }
65 return 0;
66}
67
68static size_t read_callback(void *ptr, size_t size, size_t nmemb, void *userp)
69{
70 struct WriteThis *pooh = (struct WriteThis *)userp;
71 const char *data;
72
73 if(size*nmemb < 1)
74 return 0;
75
76 data = post[pooh->counter];
77
78 if(data) {
79 size_t len = strlen(data);
80 memcpy(ptr, data, len);
81 pooh->counter++; /* advance pointer */
82 return len;
83 }
84 return 0; /* no more data left to deliver */
85}

87int test(char *URL)
88{
89 CURL *curl;
90 CURLcode res = CURLE_OK;
91 struct curl_slist *slist = NULL;
92 struct WriteThis pooh;
93 pooh.counter = 0;
94
95 if(curl_global_init(CURL_GLOBAL_ALL) != CURLE_OK) {
96 fprintf(stderr, "curl_global_init() failed\n");
97 return TEST_ERR_MAJOR_BAD;
98 }
99
100 curl = curl_easy_init();
101 if(!curl) {
102 fprintf(stderr, "curl_easy_init() failed\n");
103 curl_global_cleanup();
104 return TEST_ERR_MAJOR_BAD;
105 }
106
107 slist = curl_slist_append(slist, "Transfer-Encoding: chunked");
108 if(slist == NULL) {
109 fprintf(stderr, "curl_slist_append() failed\n");
110 curl_easy_cleanup(curl);
111 curl_global_cleanup();
112 return TEST_ERR_MAJOR_BAD;
113 }
114
115 /* First set the URL that is about to receive our POST. */
116 test_setopt(curl, CURLOPT_URL, URL);
117
118 /* Now specify we want to POST data */
119 test_setopt(curl, CURLOPT_POST, 1L);
120
121#ifdef CURL_DOES_CONVERSIONS
122 /* Convert the POST data to ASCII */
123 test_setopt(curl, CURLOPT_TRANSFERTEXT, 1L);
124#endif
125
126 /* we want to use our own read function */
127 test_setopt(curl, CURLOPT_READFUNCTION, read_callback);
128
129 /* pointer to pass to our read function */
130 test_setopt(curl, CURLOPT_READDATA, &pooh);
131
132 /* get verbose debug output please */
133 test_setopt(curl, CURLOPT_VERBOSE, 1L);
134
135 /* include headers in the output */
136 test_setopt(curl, CURLOPT_HEADER, 1L);
137
138 /* enforce chunked transfer by setting the header */
139 test_setopt(curl, CURLOPT_HTTPHEADER, slist);
140
141 test_setopt(curl, CURLOPT_HTTPAUTH, (long)CURLAUTH_DIGEST);
142 test_setopt(curl, CURLOPT_USERPWD, "foo:bar");
143
144 /* we want to use our own progress function */
145 test_setopt(curl, CURLOPT_NOPROGRESS, 0L);
146 test_setopt(curl, CURLOPT_PROGRESSFUNCTION, progress_callback);
147
148 /* Perform the request, res will get the return code */
149 res = curl_easy_perform(curl);
150
151test_cleanup:
152
153 /* clean up the headers list */
154 if(slist)
155 curl_slist_free_all(slist);
156
157 /* always cleanup */
158 curl_easy_cleanup(curl);
159 curl_global_cleanup();
160
161 return res;
162}
