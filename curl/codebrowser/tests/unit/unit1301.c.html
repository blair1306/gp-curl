22#include "curlcheck.h"
23
24#include "strcase.h"
25
26static CURLcode unit_setup(void) {return CURLE_OK;}
27static void unit_stop(void) {}
28
29UNITTEST_START
30
31int rc;
32
33rc = curl_strequal("iii", "III");
34fail_unless(rc != 0, "return code should be non-zero");
35
36rc = curl_strequal("iiia", "III");
37fail_unless(rc == 0, "return code should be zero");
38
39rc = curl_strequal("iii", "IIIa");
40fail_unless(rc == 0, "return code should be zero");
41
42rc = curl_strequal("iiiA", "IIIa");
43fail_unless(rc != 0, "return code should be non-zero");
44
45rc = curl_strnequal("iii", "III", 3);
46fail_unless(rc != 0, "return code should be non-zero");
47
48rc = curl_strnequal("iiiABC", "IIIcba", 3);
49fail_unless(rc != 0, "return code should be non-zero");
50
51rc = curl_strnequal("ii", "II", 3);
52fail_unless(rc != 0, "return code should be non-zero");
53
54UNITTEST_STOP
