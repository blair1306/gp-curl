22#include "curlcheck.h"

24#include "urldata.h"
25#include "curl_ntlm_core.h"

27static CURL *easy;
28
29static CURLcode unit_setup(void)
30{
31 int res = CURLE_OK;
32
33 global_init(CURL_GLOBAL_ALL);
34 easy = curl_easy_init();
35 if(!easy)
36 return CURLE_OUT_OF_MEMORY;
37 return res;
38}
39
40static void unit_stop(void)
41{
42 curl_easy_cleanup(easy);
43 curl_global_cleanup();
44}
