31struct fileinfo *Curl_fileinfo_alloc(void)
32{
33 return calloc(1, sizeof(struct fileinfo));
34}
35
36void Curl_fileinfo_dtor(void *user, void *element)
37{
38 struct fileinfo *finfo = element;
39 (void) user;
40 if(!finfo)
41 return;
42
43 Curl_safefree(finfo->info.b_data);
44
45 free(finfo);
46}
