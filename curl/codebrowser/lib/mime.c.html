135/*
136 * get_vms_file_size does what it takes to get the real size of the file
137 *
138 * For fixed files, find out the size of the EOF block and adjust.
139 *
140 * For all others, have to read the entire file in, discarding the contents.
141 * Most posted text files will be small, and binary files like zlib archives
142 * and CD/DVD images should be either a STREAM_LF format or a fixed format.
143 *
144 */
145curl_off_t VmsRealFileSize(const char *name,
146 const struct_stat *stat_buf)
147{
148 char buffer[8192];
149 curl_off_t count;
150 int ret_stat;
151 FILE * file;
152
153 file = fopen(name, FOPEN_READTEXT); /* VMS */
154 if(file == NULL)
155 return 0;
156
157 count = 0;
158 ret_stat = 1;
159 while(ret_stat > 0) {
160 ret_stat = fread(buffer, 1, sizeof(buffer), file);
161 if(ret_stat != 0)
162 count += ret_stat;
163 }
164 fclose(file);
165
166 return count;
167}

221/*
222 (Quote from The Open Group Base Specifications Issue 6 IEEE Std 1003.1, 2004
223 Edition)
224
225 The basename() function shall take the pathname pointed to by path and
226 return a pointer to the final component of the pathname, deleting any
227 trailing '/' characters.
228
229 If the string pointed to by path consists entirely of the '/' character,
230 basename() shall return a pointer to the string "/". If the string pointed
231 to by path is exactly "//", it is implementation-defined whether '/' or "//"
232 is returned.
233
234 If path is a null pointer or points to an empty string, basename() shall
235 return a pointer to the string ".".
236
237 The basename() function may modify the string pointed to by path, and may
238 return a pointer to static storage that may then be overwritten by a
239 subsequent call to basename().
240
241 The basename() function need not be reentrant. A function that is not
242 required to be reentrant is not required to be thread-safe.
243
244*/
245static char *Curl_basename(char *path)
246{
247 /* Ignore all the details above for now and make a quick and simple
248 implementaion here */
249 char *s1;
250 char *s2;
251
252 s1 = strrchr(path, '/');
253 s2 = strrchr(path, '\\');
254
255 if(s1 && s2) {
256 path = (s1 > s2? s1 : s2) + 1;
257 }
258 else if(s1)
259 path = s1 + 1;
260 else if(s2)
261 path = s2 + 1;
262
263 return path;
264}

279/* Escape header string into allocated memory. */
280static char *escape_string(const char *src)
281{
282 size_t bytecount = 0;
283 size_t i;
284 char *dst;
285
286 for(i = 0; src[i]; i++)
287 if(src[i] == '"' || src[i] == '\\')
288 bytecount++;
289
290 bytecount += i;
291 dst = malloc(bytecount + 1);
292 if(!dst)
293 return NULL;
294
295 for(i = 0; *src; src++) {
296 if(*src == '"' || *src == '\\')
297 dst[i++] = '\\';
298 dst[i++] = *src;
299 }
300
301 dst[i] = '\0';
302 return dst;
303}

319 size_t len = strlen(hdr);
320 char *value = NULL;
321
322 for(; !value && hdrlist; hdrlist = hdrlist->next)
323 value = match_header(hdrlist, hdr, len);

325 return value;

398/* Base64 content encoder. */
399static size_t encoder_base64_read(char *buffer, size_t size, bool ateof,
400 curl_mimepart *part)
401{
402 mime_encoder_state *st = &part->encstate;
403 size_t cursize = 0;
404 int i;
405 char *ptr = buffer;
406
407 while(st->bufbeg < st->bufend) {
408 /* Line full ? */
409 if(st->pos >= MAX_ENCODED_LINE_LENGTH - 4) {
410 /* Yes, we need 2 characters for CRLF. */
411 if(size < 2)
412 break;
413 *ptr++ = '\r';
414 *ptr++ = '\n';
415 st->pos = 0;
416 cursize += 2;
417 size -= 2;
418 }
419
420 /* Be sure there is enough space and input data for a base64 group. */
421 if(size < 4 || st->bufend - st->bufbeg < 3)
422 break;
423
424 /* Encode three bytes a four characters. */
425 i = st->buf[st->bufbeg++] & 0xFF;
426 i = (i << 8) | (st->buf[st->bufbeg++] & 0xFF);
427 i = (i << 8) | (st->buf[st->bufbeg++] & 0xFF);
428 *ptr++ = base64[(i >> 18) & 0x3F];
429 *ptr++ = base64[(i >> 12) & 0x3F];
430 *ptr++ = base64[(i >> 6) & 0x3F];
431 *ptr++ = base64[i & 0x3F];
432 cursize += 4;
433 st->pos += 4;
434 size -= 4;
435 }
436
437 /* If at eof, we have to flush the buffered data. */
438 if(ateof && size >= 4) {
439 /* Buffered data size can only be 0, 1 or 2. */
440 ptr[2] = ptr[3] = '=';
441 i = 0;
442 switch(st->bufend - st->bufbeg) {
443 case 2:
444 i = (st->buf[st->bufbeg + 1] & 0xFF) << 8;
445 /* FALLTHROUGH */
446 case 1:
447 i |= (st->buf[st->bufbeg] & 0xFF) << 16;
448 ptr[0] = base64[(i >> 18) & 0x3F];
449 ptr[1] = base64[(i >> 12) & 0x3F];
450 if(++st->bufbeg != st->bufend) {
451 ptr[2] = base64[(i >> 6) & 0x3F];
452 st->bufbeg++;
453 }
454 cursize += 4;
455 st->pos += 4;
456 break;
457 }
458 }
459
460#ifdef CURL_DOES_CONVERSIONS
461 /* This is now textual data, Convert character codes. */
462 if(part->easy && cursize) {
463 CURLcode result = Curl_convert_to_network(part->easy, buffer, cursize);
464 if(result)
465 return READ_ERROR;
466 }
467#endif
468
469 return cursize;
470}

472static curl_off_t encoder_base64_size(curl_mimepart *part)
473{
474 curl_off_t size = part->datasize;
475
476 if(size <= 0)
477 return size; /* Unknown size or no data. */
478
479 /* Compute base64 character count. */
480 size = 4 * (1 + (size - 1) / 3);
481
482 /* Effective character count must include CRLFs. */
483 return size + 2 * ((size - 1) / MAX_ENCODED_LINE_LENGTH);
484}

487/* Quoted-printable lookahead.
488 *
489 * Check if a CRLF or end of data is in input buffer at current position + n.
490 * Return -1 if more data needed, 1 if CRLF or end of data, else 0.
491 */
492static int qp_lookahead_eol(mime_encoder_state *st, int ateof, size_t n)
493{
494 n += st->bufbeg;
495 if(n >= st->bufend && ateof)
496 return 1;
497 if(n + 2 > st->bufend)
498 return ateof? 0: -1;
499 if(qp_class[st->buf[n] & 0xFF] == QP_CR &&
500 qp_class[st->buf[n + 1] & 0xFF] == QP_LF)
501 return 1;
502 return 0;
503}

505/* Quoted-printable encoder. */
506static size_t encoder_qp_read(char *buffer, size_t size, bool ateof,
507 curl_mimepart *part)
508{
509 mime_encoder_state *st = &part->encstate;
510 char *ptr = buffer;
511 size_t cursize = 0;
512 int i;
513 size_t len;
514 size_t consumed;
515 int softlinebreak;
516 char buf[4];
517
518 /* On all platforms, input is supposed to be ASCII compatible: for this
519 reason, we use hexadecimal ASCII codes in this function rather than
520 character constants that can be interpreted as non-ascii on some
521 platforms. Preserve ASCII encoding on output too. */
522 while(st->bufbeg < st->bufend) {
523 len = 1;
524 consumed = 1;
525 i = st->buf[st->bufbeg];
526 buf[0] = (char) i;
527 buf[1] = aschex[(i >> 4) & 0xF];
528 buf[2] = aschex[i & 0xF];
529
530 switch(qp_class[st->buf[st->bufbeg] & 0xFF]) {
531 case QP_OK: /* Not a special character. */
532 break;
533 case QP_SP: /* Space or tab. */
534 /* Spacing must be escaped if followed by CRLF. */
535 switch(qp_lookahead_eol(st, ateof, 1)) {
536 case -1: /* More input data needed. */
537 return cursize;
538 case 0: /* No encoding needed. */
539 break;
540 default: /* CRLF after space or tab. */
541 buf[0] = '\x3D'; /* '=' */
542 len = 3;
543 break;
544 }
545 break;
546 case QP_CR: /* Carriage return. */
547 /* If followed by a line-feed, output the CRLF pair.
548 Else escape it. */
549 switch(qp_lookahead_eol(st, ateof, 0)) {
550 case -1: /* Need more data. */
551 return cursize;
552 case 1: /* CRLF found. */
553 buf[len++] = '\x0A'; /* Append '\n'. */
554 consumed = 2;
555 break;
556 default: /* Not followed by LF: escape. */
557 buf[0] = '\x3D'; /* '=' */
558 len = 3;
559 break;
560 }
561 break;
562 default: /* Character must be escaped. */
563 buf[0] = '\x3D'; /* '=' */
564 len = 3;
565 break;
566 }
567
568 /* Be sure the encoded character fits within maximum line length. */
569 if(buf[len - 1] != '\x0A') { /* '\n' */
570 softlinebreak = st->pos + len > MAX_ENCODED_LINE_LENGTH;
571 if(!softlinebreak && st->pos + len == MAX_ENCODED_LINE_LENGTH) {
572 /* We may use the current line only if end of data or followed by
573 a CRLF. */
574 switch(qp_lookahead_eol(st, ateof, consumed)) {
575 case -1: /* Need more data. */
576 return cursize;
577 break;
578 case 0: /* Not followed by a CRLF. */
579 softlinebreak = 1;
580 break;
581 }
582 }
583 if(softlinebreak) {
584 strcpy(buf, "\x3D\x0D\x0A"); /* "=\r\n" */
585 len = 3;
586 consumed = 0;
587 }
588 }
589
590 /* If the output buffer would overflow, do not store. */
591 if(len > size)
592 break;
593
594 /* Append to output buffer. */
595 memcpy(ptr, buf, len);
596 cursize += len;
597 ptr += len;
598 size -= len;
599 st->pos += len;
600 if(buf[len - 1] == '\x0A') /* '\n' */
601 st->pos = 0;
602 st->bufbeg += consumed;
603 }
604
605 return cursize;
606}

685static int mime_file_seek(void *instream, curl_off_t offset, int whence)
686{
687 curl_mimepart *part = (curl_mimepart *) instream;
688
689 if(whence == SEEK_SET && !offset && !part->fp)
690 return CURL_SEEKFUNC_OK; /* Not open: implicitly already at BOF. */
691
692 if(mime_open_file(part))
693 return CURL_SEEKFUNC_FAIL;
694

715/* Readback a byte string segment. */
716static size_t readback_bytes(mime_state *state,
717 char *buffer, size_t bufsize,
718 const char *bytes, size_t numbytes,
719 const char *trail)
720{
721 size_t sz;
722
723 sz = numbytes - state->offset;
724
725 if(numbytes > state->offset) {
726 sz = numbytes - state->offset;
727 bytes += state->offset;
728 }
729 else {
730 size_t tsz = strlen(trail);
731
732 sz = state->offset - numbytes;
733 if(sz >= tsz)
734 return 0;
735 bytes = trail + sz;
736 sz = tsz - sz;
737 }
738
739 if(sz > bufsize)
740 sz = bufsize;
741
742 memcpy(buffer, bytes, sz);
743 state->offset += sz;
744 return sz;
745}

749 char *buffer, size_t bufsize)

758/* Read and encode part content. */
759static size_t read_encoded_part_content(curl_mimepart *part,
760 char *buffer, size_t bufsize)
761{
762 mime_encoder_state *st = &part->encstate;
763 size_t cursize = 0;
764 size_t sz;
765 bool ateof = FALSE;
766
767 while(bufsize) {
768 if(st->bufbeg < st->bufend || ateof) {
769 /* Encode buffered data. */
770 sz = part->encoder->encodefunc(buffer, bufsize, ateof, part);
771 switch(sz) {
772 case 0:
773 if(ateof)
774 return cursize;
775 break;
776 case CURL_READFUNC_ABORT:
777 case CURL_READFUNC_PAUSE:
778 case READ_ERROR:
779 return cursize? cursize: sz;
780 default:
781 cursize += sz;
782 buffer += sz;
783 bufsize -= sz;
784 continue;
785 }
786 }
787
788 /* We need more data in input buffer. */
789 if(st->bufbeg) {
790 size_t len = st->bufend - st->bufbeg;
791
792 if(len)
793 memmove(st->buf, st->buf + st->bufbeg, len);
794 st->bufbeg = 0;
795 st->bufend = len;
796 }
797 if(st->bufend >= sizeof st->buf)
798 return cursize? cursize: READ_ERROR; /* Buffer full. */
799 sz = read_part_content(part, st->buf + st->bufend,
800 sizeof st->buf - st->bufend);
801 switch(sz) {
802 case 0:
803 ateof = TRUE;
804 break;
805 case CURL_READFUNC_ABORT:
806 case CURL_READFUNC_PAUSE:
807 case READ_ERROR:
808 return cursize? cursize: sz;
809 default:
810 st->bufend += sz;
811 break;
812 }
813 }
814
815 return cursize;
816}

818/* Readback a mime part. */
819static size_t readback_part(curl_mimepart *part,
820 char *buffer, size_t bufsize)
821{
822 size_t cursize = 0;
823 size_t sz;
824 struct curl_slist *hdr;
825#ifdef CURL_DOES_CONVERSIONS
826 char *convbuf = buffer;
827#endif
828
829 /* Readback from part. */
830
831 while(bufsize) {
832 sz = 0;
833 hdr = (struct curl_slist *) part->state.ptr;
834 switch(part->state.state) {
835 case MIMESTATE_BEGIN:
836 mimesetstate(&part->state, part->flags & MIME_BODY_ONLY? MIMESTATE_BODY:
837 MIMESTATE_CURLHEADERS, part->curlheaders);
838 break;
839 case MIMESTATE_USERHEADERS:
840 if(!hdr) {
841 mimesetstate(&part->state, MIMESTATE_EOH, NULL);
842 break;
843 }
844 if(match_header(hdr, "Content-Type", 12)) {
845 mimesetstate(&part->state, MIMESTATE_USERHEADERS, hdr->next);
846 break;
847 }
848 /* FALLTHROUGH */
849 case MIMESTATE_CURLHEADERS:
850 if(!hdr)
851 mimesetstate(&part->state, MIMESTATE_USERHEADERS, part->userheaders);
852 else {
853 sz = readback_bytes(&part->state, buffer, bufsize,
854 hdr->data, strlen(hdr->data), "\r\n");
855 if(!sz)
856 mimesetstate(&part->state, part->state.state, hdr->next);
857 }
858 break;
859 case MIMESTATE_EOH:
860 sz = readback_bytes(&part->state, buffer, bufsize, "\r\n", 2, "");
861 if(!sz)
862 mimesetstate(&part->state, MIMESTATE_BODY, NULL);
863 break;
864 case MIMESTATE_BODY:
865#ifdef CURL_DOES_CONVERSIONS
866 if(part->easy && convbuf < buffer) {
867 CURLcode result = Curl_convert_to_network(part->easy, convbuf,
868 buffer - convbuf);
869 if(result)
870 return READ_ERROR;
871 convbuf = buffer;
872 }
873#endif
874 cleanup_encoder_state(&part->encstate);
875 mimesetstate(&part->state, MIMESTATE_CONTENT, NULL);
876 break;
877 case MIMESTATE_CONTENT:
878 if(part->encoder)
879 sz = read_encoded_part_content(part, buffer, bufsize);
880 else
881 sz = read_part_content(part, buffer, bufsize);
882 switch(sz) {
883 case 0:
884 mimesetstate(&part->state, MIMESTATE_END, NULL);
885 /* Try sparing open file descriptors. */
886 if(part->kind == MIMEKIND_FILE && part->fp) {
887 fclose(part->fp);
888 part->fp = NULL;
889 }
890 /* FALLTHROUGH */
891 case CURL_READFUNC_ABORT:
892 case CURL_READFUNC_PAUSE:
893 case READ_ERROR:
894 return cursize? cursize: sz;
895 }
896 break;
897 case MIMESTATE_END:
898 return cursize;
899 default:
900 break; /* Other values not in part state. */
901 }
902
903 /* Bump buffer and counters according to read size. */
904 cursize += sz;
905 buffer += sz;
906 bufsize -= sz;
907 }
908
909#ifdef CURL_DOES_CONVERSIONS
910 if(part->easy && convbuf < buffer &&
911 part->state.state < MIMESTATE_BODY) {
912 CURLcode result = Curl_convert_to_network(part->easy, convbuf,
913 buffer - convbuf);
914 if(result)
915 return READ_ERROR;
916 }
917#endif
918
919 return cursize;
920}

922/* Readback from mime. */
923static size_t mime_subparts_read(char *buffer, size_t size, size_t nitems,
924 void *instream)
925{
926 curl_mime *mime = (curl_mime *) instream;
927 size_t cursize = 0;
928 size_t sz;
929 curl_mimepart *part;
930#ifdef CURL_DOES_CONVERSIONS
931 char *convbuf = buffer;
932#endif
933
934 (void) size; /* Always 1. */
935
936 while(nitems) {
937 sz = 0;
938 part = mime->state.ptr;
939 switch(mime->state.state) {
940 case MIMESTATE_BEGIN:
941 case MIMESTATE_BODY:
942#ifdef CURL_DOES_CONVERSIONS
943 convbuf = buffer;
944#endif
945 mimesetstate(&mime->state, MIMESTATE_BOUNDARY1, mime->firstpart);
946 /* The first boundary always follows the header termination empty line,
947 so is always preceded by a CRLK. We can then spare 2 characters
948 by skipping the leading CRLF in boundary. */
949 mime->state.offset += 2;
950 break;
951 case MIMESTATE_BOUNDARY1:
952 sz = readback_bytes(&mime->state, buffer, nitems, "\r\n--", 4, "");
953 if(!sz)
954 mimesetstate(&mime->state, MIMESTATE_BOUNDARY2, part);
955 break;
956 case MIMESTATE_BOUNDARY2:
957 sz = readback_bytes(&mime->state, buffer, nitems, mime->boundary,
958 strlen(mime->boundary), part? "\r\n": "--\r\n");
959 if(!sz) {
960#ifdef CURL_DOES_CONVERSIONS
961 if(mime->easy && convbuf < buffer) {
962 CURLcode result = Curl_convert_to_network(mime->easy, convbuf,
963 buffer - convbuf);
964 if(result)
965 return READ_ERROR;
966 convbuf = buffer;
967 }
968#endif
969 mimesetstate(&mime->state, MIMESTATE_CONTENT, part);
970 }
971 break;
972 case MIMESTATE_CONTENT:
973 if(!part) {
974 mimesetstate(&mime->state, MIMESTATE_END, NULL);
975 break;
976 }
977 sz = readback_part(part, buffer, nitems);
978 switch(sz) {
979 case CURL_READFUNC_ABORT:
980 case CURL_READFUNC_PAUSE:
981 case READ_ERROR:
982 return cursize? cursize: sz;
983 case 0:
984#ifdef CURL_DOES_CONVERSIONS
985 convbuf = buffer;
986#endif
987 mimesetstate(&mime->state, MIMESTATE_BOUNDARY1, part->nextpart);
988 break;
989 }
990 break;
991 case MIMESTATE_END:
992 return cursize;
993 default:
994 break; /* other values not used in mime state. */
995 }
996
997 /* Bump buffer and counters according to read size. */
998 cursize += sz;
999 buffer += sz;
1000 nitems -= sz;
1001 }
1002
1003#ifdef CURL_DOES_CONVERSIONS
1004 if(mime->easy && convbuf < buffer &&
1005 mime->state.state <= MIMESTATE_CONTENT) {
1006 CURLcode result = Curl_convert_to_network(mime->easy, convbuf,
1007 buffer - convbuf);
1008 if(result)
1009 return READ_ERROR;
1010 }
1011#endif
1012
1013 return cursize;
1014}

1016static int mime_part_rewind(curl_mimepart *part)
1017{
1018 int res = CURL_SEEKFUNC_OK;
1019 enum mimestate targetstate = MIMESTATE_BEGIN;
1020
1021 if(part->flags & MIME_BODY_ONLY)
1022 targetstate = MIMESTATE_BODY;
1023 cleanup_encoder_state(&part->encstate);
1024 if(part->state.state > targetstate) {
1025 res = CURL_SEEKFUNC_CANTSEEK;
1026 if(part->seekfunc) {
1027 res = part->seekfunc(part->arg, (curl_off_t) 0, SEEK_SET);
1028 switch(res) {
1029 case CURL_SEEKFUNC_OK:
1030 case CURL_SEEKFUNC_FAIL:
1031 case CURL_SEEKFUNC_CANTSEEK:
1032 break;
1033 case -1: /* For fseek() error. */
1034 res = CURL_SEEKFUNC_CANTSEEK;
1035 break;
1036 default:
1037 res = CURL_SEEKFUNC_FAIL;
1038 break;
1039 }
1040 }
1041 }
1042
1043 if(res == CURL_SEEKFUNC_OK)
1044 mimesetstate(&part->state, targetstate, NULL);
1045
1046 return res;
1047}

1049static int mime_subparts_seek(void *instream, curl_off_t offset, int whence)
1050{
1051 curl_mime *mime = (curl_mime *) instream;
1052 curl_mimepart *part;
1053 int result = CURL_SEEKFUNC_OK;
1054 int res;
1055
1056 if(whence != SEEK_SET || offset)
1057 return CURL_SEEKFUNC_CANTSEEK; /* Only support full rewind. */
1058
1059 if(mime->state.state == MIMESTATE_BEGIN)
1060 return CURL_SEEKFUNC_OK; /* Already rewound. */
1061
1062 for(part = mime->firstpart; part; part = part->nextpart) {
1063 res = mime_part_rewind(part);
1064 if(res != CURL_SEEKFUNC_OK)
1065 result = res;
1066 }
1067
1068 if(result == CURL_SEEKFUNC_OK)
1069 mimesetstate(&mime->state, MIMESTATE_BEGIN, NULL);
1070
1071 return result;
1072}

1411/* Set mime part content from subparts. */
1412CURLcode Curl_mime_set_subparts(curl_mimepart *part,
1413 curl_mime *subparts, int take_ownership)
1414{
1415 curl_mime *root;
1416
1417 if(!part)
1418 return CURLE_BAD_FUNCTION_ARGUMENT;
1419
1420 /* Accept setting twice the same subparts. */
1421 if(part->kind == MIMEKIND_MULTIPART && part->arg == subparts)
1422 return CURLE_OK;
1423
1424 cleanup_part_content(part);
1425
1426 if(subparts) {
1427 /* Must belong to the same data handle. */
1428 if(part->easy && subparts->easy && part->easy != subparts->easy)
1429 return CURLE_BAD_FUNCTION_ARGUMENT;
1430
1431 /* Should not have been attached already. */
1432 if(subparts->parent)
1433 return CURLE_BAD_FUNCTION_ARGUMENT;
1434
1435 /* Should not be the part's root. */
1436 root = part->parent;
1437 if(root) {
1438 while(root->parent && root->parent->parent)
1439 root = root->parent->parent;
1440 if(subparts == root) {
1441 failf(part->easy, "Can't add itself as a subpart!");
1442 return CURLE_BAD_FUNCTION_ARGUMENT;
1443 }
1444 }
1445
1446 subparts->parent = part;
1447 part->readfunc = mime_subparts_read;
1448 part->seekfunc = mime_subparts_seek;
1449 part->freefunc = take_ownership? mime_subparts_free: mime_subparts_unbind;
1450 part->arg = subparts;
1451 part->datasize = -1;
1452 part->kind = MIMEKIND_MULTIPART;
1453 }
1454
1455 return CURLE_OK;
1456}

1482static size_t slist_size(struct curl_slist *s,
1483 size_t overhead, const char *skip)
1484{

1486 size_t skiplen = skip? strlen(skip): 0;
1487

1494/* Get/compute multipart size. */
1495static curl_off_t multipart_size(curl_mime *mime)
1496{
1497 curl_off_t size;
1498 curl_off_t sz;
1499 size_t boundarysize;
1500 curl_mimepart *part;
1501
1502 if(!mime)
1503 return 0; /* Not present -> empty. */
1504
1505 boundarysize = 4 + strlen(mime->boundary) + 2;
1506 size = boundarysize; /* Final boundary - CRLF after headers. */
1507
1508 for(part = mime->firstpart; part; part = part->nextpart) {
1509 sz = Curl_mime_size(part);
1510
1511 if(sz < 0)
1512 size = sz;
1513
1514 if(size >= 0)
1515 size += boundarysize + sz;
1516 }
1517
1518 return size;
1519}

1521/* Get/compute mime size. */
1522curl_off_t Curl_mime_size(curl_mimepart *part)
1523{
1524 curl_off_t size;
1525
1526 if(part->datasize < 0 && part->kind == MIMEKIND_MULTIPART)
1527 part->datasize = multipart_size(part->arg);
1528
1529 size = part->datasize;
1530
1531 if(part->encoder)
1532 size = part->encoder->sizefunc(part);
1533
1534 if(size >= 0 && !(part->flags & MIME_BODY_ONLY)) {
1535 /* Compute total part size. */
1536 size += slist_size(part->curlheaders, 2, NULL);
1537 size += slist_size(part->userheaders, 2, "Content-Type");
1538 size += 2; /* CRLF after headers. */
1539 }
1540 return size;
1541}

1576static const char *ContentTypeForFilename(const char *filename)
1577{
1578 unsigned int i;
1579
1580 /*
1581 * If no content type was specified, we scan through a few well-known
1582 * extensions and pick the first we match!
1583 */
1584 struct ContentType {
1585 const char *extension;
1586 const char *type;
1587 };
1588 static const struct ContentType ctts[] = {
1589 {".gif", "image/gif"},
1590 {".jpg", "image/jpeg"},
1591 {".jpeg", "image/jpeg"},
1592 {".png", "image/png"},
1593 {".svg", "image/svg+xml"},
1594 {".txt", "text/plain"},
1595 {".htm", "text/html"},
1596 {".html", "text/html"},
1597 {".pdf", "application/pdf"},
1598 {".xml", "application/xml"}
1599 };
1600
1601 if(filename) {
1602 size_t len1 = strlen(filename);
1603 const char *nameend = filename + len1;
1604
1605 for(i = 0; i < sizeof ctts / sizeof ctts[0]; i++) {
1606 size_t len2 = strlen(ctts[i].extension);
1607
1608 if(len1 >= len2 && strcasecompare(nameend - len2, ctts[i].extension))
1609 return ctts[i].type;
1610 }
1611 }
1612 return NULL;
1613}

1615CURLcode Curl_mime_prepare_headers(curl_mimepart *part,
1616 const char *contenttype,
1617 const char *disposition,
1618 enum mimestrategy strategy)
1619{
1620 curl_mime *mime = NULL;
1621 const char *boundary = NULL;
1622 char *s;
1623 const char *cte = NULL;
1624 CURLcode ret = CURLE_OK;
1625
1626 /* Get rid of previously prepared headers. */
1627 curl_slist_free_all(part->curlheaders);
1628 part->curlheaders = NULL;
1629
1630 /* Be sure we won't access old headers later. */
1631 if(part->state.state == MIMESTATE_CURLHEADERS)
1632 mimesetstate(&part->state, MIMESTATE_CURLHEADERS, NULL);
1633
1634 /* Build the content-type header. */
1635 s = search_header(part->userheaders, "Content-Type");
1636 if(s)
1637 contenttype = s;
1638 if(part->mimetype)
1639 contenttype = part->mimetype;
1640 if(!contenttype) {
1641 switch(part->kind) {
1642 case MIMEKIND_MULTIPART:
1643 contenttype = MULTIPART_CONTENTTYPE_DEFAULT;
1644 break;
1645 case MIMEKIND_FILE:
1646 contenttype = ContentTypeForFilename(part->filename);
1647 if(!contenttype)
1648 contenttype = ContentTypeForFilename(part->data);
1649 if(!contenttype && part->filename)
1650 contenttype = FILE_CONTENTTYPE_DEFAULT;
1651 break;
1652 default:
1653 contenttype = ContentTypeForFilename(part->filename);
1654 break;
1655 }
1656 }
1657
1658 if(part->kind == MIMEKIND_MULTIPART) {
1659 mime = (curl_mime *) part->arg;
1660 if(mime)
1661 boundary = mime->boundary;
1662 }
1663 else if(contenttype && strcasecompare(contenttype, "text/plain"))
1664 if(strategy == MIMESTRATEGY_MAIL || !part->filename)
1665 contenttype = NULL;
1666
1667 /* Issue content-disposition header only if not already set by caller. */
1668 if(!search_header(part->userheaders, "Content-Disposition")) {
1669 if(!disposition)
1670 if(part->filename || part->name ||
1671 (contenttype && !strncasecompare(contenttype, "multipart/", 10)))
1672 disposition = DISPOSITION_DEFAULT;
1673 if(disposition && curl_strequal(disposition, "attachment") &&
1674 !part->name && !part->filename)
1675 disposition = NULL;
1676 if(disposition) {
1677 char *name = NULL;
1678 char *filename = NULL;
1679
1680 if(part->name) {
1681 name = escape_string(part->name);
1682 if(!name)
1683 ret = CURLE_OUT_OF_MEMORY;
1684 }
1685 if(!ret && part->filename) {
1686 filename = escape_string(part->filename);
1687 if(!filename)
1688 ret = CURLE_OUT_OF_MEMORY;
1689 }
1690 if(!ret)
1691 ret = Curl_mime_add_header(&part->curlheaders,
1692 "Content-Disposition: %s%s%s%s%s%s%s",
1693 disposition,
1694 name? "; name=\"": "",
1695 name? name: "",
1696 name? "\"": "",
1697 filename? "; filename=\"": "",
1698 filename? filename: "",
1699 filename? "\"": "");
1700 Curl_safefree(name);
1701 Curl_safefree(filename);
1702 if(ret)
1703 return ret;
1704 }
1705 }
1706
1707 /* Issue Content-Type header. */
1708 if(contenttype) {
1709 ret = add_content_type(&part->curlheaders, contenttype, boundary);
1710 if(ret)
1711 return ret;
1712 }
1713
1714 /* Content-Transfer-Encoding header. */
1715 if(!search_header(part->userheaders, "Content-Transfer-Encoding")) {
1716 if(part->encoder)
1717 cte = part->encoder->name;
1718 else if(contenttype && strategy == MIMESTRATEGY_MAIL &&
1719 part->kind != MIMEKIND_MULTIPART)
1720 cte = "8bit";
1721 if(cte) {
1722 ret = Curl_mime_add_header(&part->curlheaders,
1723 "Content-Transfer-Encoding: %s", cte);
1724 if(ret)
1725 return ret;
1726 }
1727 }
1728
1729 /* If we were reading curl-generated headers, restart with new ones (this
1730 should not occur). */
1731 if(part->state.state == MIMESTATE_CURLHEADERS)
1732 mimesetstate(&part->state, MIMESTATE_CURLHEADERS, part->curlheaders);
1733
1734 /* Process subparts. */
1735 if(part->kind == MIMEKIND_MULTIPART && mime) {
1736 curl_mimepart *subpart;
1737
1738 disposition = NULL;
1739 if(strcasecompare(contenttype, "multipart/form-data"))
1740 disposition = "form-data";
1741 for(subpart = mime->firstpart; subpart; subpart = subpart->nextpart) {
1742 ret = Curl_mime_prepare_headers(subpart, NULL, disposition, strategy);
1743 if(ret)
1744 return ret;
1745 }
1746 }
1747 return ret;
1748}
