106/* always use this function to change state, to make debugging easier */
107static void mstate(struct Curl_easy *data, CURLMstate state
108#ifdef DEBUGBUILD
109 , int lineno
110#endif
111)
112{
113 CURLMstate oldstate = data->mstate;
114 static const init_multistate_func finit[CURLM_STATE_LAST] = {
115 NULL,
116 NULL,
117 Curl_init_CONNECT, /* CONNECT */
118 NULL,
119 NULL,
120 NULL,
121 NULL,
122 NULL,
123 NULL,
124 Curl_connect_free /* DO */
125 /* the rest is NULL too */
126 };
127
128#if defined(DEBUGBUILD) && defined(CURL_DISABLE_VERBOSE_STRINGS)
129 (void) lineno;
130#endif
131
132 if(oldstate == state)
133 /* don't bother when the new state is the same as the old state */
134 return;
135
136 data->mstate = state;
137
138#if defined(DEBUGBUILD) && !defined(CURL_DISABLE_VERBOSE_STRINGS)
139 if(data->mstate >= CURLM_STATE_CONNECT_PEND &&
140 data->mstate < CURLM_STATE_COMPLETED) {
141 long connection_id = -5000;
142
143 if(data->easy_conn)
144 connection_id = data->easy_conn->connection_id;
145
146 infof(data,
147 "STATE: %s => %s handle %p; line %d (connection #%ld)\n",
148 statename[oldstate], statename[data->mstate],
149 (void *)data, lineno, connection_id);
150 }
151#endif
152
153 if(state == CURLM_STATE_COMPLETED)
154 /* changing to COMPLETED means there's one less easy handle 'alive' */
155 data->multi->num_alive--;
156
157 /* if this state has an init-function, run it */
158 if(finit[state])
159 finit[state](data);
160}

172struct Curl_sh_entry {
173 struct Curl_easy *easy;
174 int action; /* what action READ/WRITE this socket waits for */
175 curl_socket_t socket; /* mainly to ease debugging */
176 void *socketp; /* settable by users with curl_multi_assign() */
177};
178/* bits for 'action' having no bits means this socket is not expecting any
179 action */
180#define SH_READ 1
181#define SH_WRITE 2
182
183/* look up a given socket in the socket hash, skip invalid sockets */
184static struct Curl_sh_entry *sh_getentry(struct curl_hash *sh,
185 curl_socket_t s)
186{
187 if(s != CURL_SOCKET_BAD)
188 /* only look for proper sockets */
189 return Curl_hash_pick(sh, (char *)&s, sizeof(curl_socket_t));
190 return NULL;
191}
192
193/* make sure this socket is present in the hash for this handle */
194static struct Curl_sh_entry *sh_addentry(struct curl_hash *sh,
195 curl_socket_t s,
196 struct Curl_easy *data)
197{
198 struct Curl_sh_entry *there = sh_getentry(sh, s);
199 struct Curl_sh_entry *check;
200
201 if(there)
202 /* it is present, return fine */
203 return there;
204
205 /* not present, add it */
206 check = calloc(1, sizeof(struct Curl_sh_entry));
207 if(!check)
208 return NULL; /* major failure */
209
210 check->easy = data;
211 check->socket = s;
212
213 /* make/add new hash entry */
214 if(!Curl_hash_add(sh, (char *)&s, sizeof(curl_socket_t), check)) {
215 free(check);
216 return NULL; /* major failure */
217 }
218
219 return check; /* things are good in sockhash land */
220}

223/* delete the given socket + handle from the hash */

226 /* We remove the hash entry. This will end up in a call to

241static size_t fd_key_compare(void *k1, size_t k1_len, void *k2, size_t k2_len)
242{
243 (void) k1_len; (void) k2_len;

248static size_t hash_fd(void *key, size_t key_length, size_t slots_num)
249{
250 curl_socket_t fd = *((curl_socket_t *) key);
251 (void) key_length;
252
253 return (fd % slots_num);

256/*
257 * sh_init() creates a new socket hash and returns the handle for it.
258 *
259 * Quote from README.multi_socket:
260 *
261 * "Some tests at 7000 and 9000 connections showed that the socket hash lookup
262 * is somewhat of a bottle neck. Its current implementation may be a bit too
263 * limiting. It simply has a fixed-size array, and on each entry in the array
264 * it has a linked list with entries. So the hash only checks which list to
265 * scan through. The code I had used so for used a list with merely 7 slots
266 * (as that is what the DNS hash uses) but with 7000 connections that would
267 * make an average of 1000 nodes in each list to run through. I upped that to
268 * 97 slots (I believe a prime is suitable) and noticed a significant speed
269 * increase. I need to reconsider the hash implementation or use a rather
270 * large default value like this. At 9000 connections I was still below 10us
271 * per call."
272 *
273 */
274static int sh_init(struct curl_hash *hash, int hashsize)
275{
276 return Curl_hash_init(hash, hashsize, hash_fd, fd_key_compare,
277 sh_freeentry);
278}

305struct Curl_multi *Curl_multi_handle(int hashsize, /* socket hash */
306 int chashsize) /* connection hash */
307{
308 struct Curl_multi *multi = calloc(1, sizeof(struct Curl_multi));
309
310 if(!multi)
311 return NULL;
312
313 multi->type = CURL_MULTI_HANDLE;
314
315 if(Curl_mk_dnscache(&multi->hostcache))
316 goto error;
317
318 if(sh_init(&multi->sockhash, hashsize))
319 goto error;
320
321 if(Curl_conncache_init(&multi->conn_cache, chashsize))
322 goto error;
323
324 Curl_llist_init(&multi->msglist, multi_freeamsg);
325 Curl_llist_init(&multi->pending, multi_freeamsg);
326
327 /* allocate a new easy handle to use when closing cached connections */
328 multi->closure_handle = curl_easy_init();
329 if(!multi->closure_handle)
330 goto error;
331
332 multi->closure_handle->multi = multi;
333 multi->closure_handle->state.conn_cache = &multi->conn_cache;
334
335 multi->max_pipeline_length = 5;
336
337 /* -1 means it not set by user, use the default value */
338 multi->maxconnects = -1;
339 return multi;
340
341 error:
342
343 Curl_hash_destroy(&multi->sockhash);
344 Curl_hash_destroy(&multi->hostcache);
345 Curl_conncache_destroy(&multi->conn_cache);
346 Curl_close(multi->closure_handle);
347 multi->closure_handle = NULL;
348 Curl_llist_destroy(&multi->msglist, NULL);
349 Curl_llist_destroy(&multi->pending, NULL);
350
351 free(multi);
352 return NULL;
353}

361CURLMcode curl_multi_add_handle(struct Curl_multi *multi,
362 struct Curl_easy *data)
363{
364 /* First, make some basic checks that the CURLM handle is a good handle */
365 if(!GOOD_MULTI_HANDLE(multi))
366 return CURLM_BAD_HANDLE;
367
368 /* Verify that we got a somewhat good easy handle too */
369 if(!GOOD_EASY_HANDLE(data))
370 return CURLM_BAD_EASY_HANDLE;
371
372 /* Prevent users from adding same easy handle more than once and prevent
373 adding to more than one multi stack */
374 if(data->multi)
375 return CURLM_ADDED_ALREADY;
376
377 /* Initialize timeout list for this handle */
378 Curl_llist_init(&data->state.timeoutlist, NULL);
379
380 /*
381 * No failure allowed in this function beyond this point. And no
382 * modification of easy nor multi handle allowed before this except for
383 * potential multi's connection cache growing which won't be undone in this
384 * function no matter what.
385 */
386
387 /* set the easy handle */
388 multistate(data, CURLM_STATE_INIT);
389
390 if((data->set.global_dns_cache) &&
391 (data->dns.hostcachetype != HCACHE_GLOBAL)) {
392 /* global dns cache was requested but still isn't */
393 struct curl_hash *global = Curl_global_host_cache_init();
394 if(global) {
395 /* only do this if the global cache init works */
396 data->dns.hostcache = global;
397 data->dns.hostcachetype = HCACHE_GLOBAL;
398 }
399 }
400 /* for multi interface connections, we share DNS cache automatically if the
401 easy handle's one is currently not set. */
402 else if(!data->dns.hostcache ||
403 (data->dns.hostcachetype == HCACHE_NONE)) {
404 data->dns.hostcache = &multi->hostcache;
405 data->dns.hostcachetype = HCACHE_MULTI;
406 }
407
408 /* Point to the multi's connection cache */
409 data->state.conn_cache = &multi->conn_cache;
410
411 /* This adds the new entry at the 'end' of the doubly-linked circular
412 list of Curl_easy structs to try and maintain a FIFO queue so
413 the pipelined requests are in order. */
414
415 /* We add this new entry last in the list. */
416
417 data->next = NULL; /* end of the line */
418 if(multi->easyp) {
419 struct Curl_easy *last = multi->easylp;
420 last->next = data;
421 data->prev = last;
422 multi->easylp = data; /* the new last node */
423 }
424 else {
425 /* first node, make prev NULL! */
426 data->prev = NULL;
427 multi->easylp = multi->easyp = data; /* both first and last */
428 }
429
430 /* make the Curl_easy refer back to this multi handle */
431 data->multi = multi;
432
433 /* Set the timeout for this handle to expire really soon so that it will
434 be taken care of even when this handle is added in the midst of operation
435 when only the curl_multi_socket() API is used. During that flow, only
436 sockets that time-out or have actions will be dealt with. Since this
437 handle has no action yet, we make sure it times out to get things to
438 happen. */
439 Curl_expire(data, 0, EXPIRE_RUN_NOW);
440
441 /* increase the node-counter */
442 multi->num_easy++;
443
444 /* increase the alive-counter */
445 multi->num_alive++;
446
447 /* A somewhat crude work-around for a little glitch in update_timer() that
448 happens if the lastcall time is set to the same time when the handle is
449 removed as when the next handle is added, as then the check in
450 update_timer() that prevents calling the application multiple times with
451 the same timer infor will not trigger and then the new handle's timeout
452 will not be notified to the app.
453
454 The work-around is thus simply to clear the 'lastcall' variable to force
455 update_timer() to always trigger a callback to the app when a new easy
456 handle is added */
457 memset(&multi->timer_lastcall, 0, sizeof(multi->timer_lastcall));
458
459 /* The closure handle only ever has default timeouts set. To improve the
460 state somewhat we clone the timeouts from each added handle so that the
461 closure handle always has the same timeouts as the most recently added
462 easy handle. */
463 multi->closure_handle->set.timeout = data->set.timeout;
464 multi->closure_handle->set.server_response_timeout =
465 data->set.server_response_timeout;
466
467 update_timer(multi);
468 return CURLM_OK;
469}

487/* Mark the connection as 'idle', or close it if the cache is full.
488 Returns TRUE if the connection is kept, or FALSE if it was closed. */
489static bool
490ConnectionDone(struct Curl_easy *data, struct connectdata *conn)
491{
492 /* data->multi->maxconnects can be negative, deal with it. */
493 size_t maxconnects =
494 (data->multi->maxconnects < 0) ? data->multi->num_easy * 4:
495 data->multi->maxconnects;
496 struct connectdata *conn_candidate = NULL;
497
498 /* Mark the current connection as 'unused' */
499 conn->inuse = FALSE;
500
501 if(maxconnects > 0 &&
502 data->state.conn_cache->num_connections > maxconnects) {
503 infof(data, "Connection cache is full, closing the oldest one.\n");
504
505 conn_candidate = Curl_oldest_idle_connection(data);
506
507 if(conn_candidate) {
508 /* Set the connection's owner correctly */
509 conn_candidate->data = data;
510
511 /* the winner gets the honour of being disconnected */
512 (void)Curl_disconnect(conn_candidate, /* dead_connection */ FALSE);
513 }
514 }
515
516 return (conn_candidate == conn) ? FALSE : TRUE;
517}

519static CURLcode multi_done(struct connectdata **connp,
520 CURLcode status, /* an error if this is called
521 after an error was detected */
522 bool premature)
523{
524 CURLcode result;
525 struct connectdata *conn;
526 struct Curl_easy *data;
527 unsigned int i;
528
529 DEBUGASSERT(*connp);
530
531 conn = *connp;
532 data = conn->data;
533
534 DEBUGF(infof(data, "multi_done\n"));
535
536 if(data->state.done)
537 /* Stop if multi_done() has already been called */
538 return CURLE_OK;
539
540 Curl_getoff_all_pipelines(data, conn);
541
542 /* Cleanup possible redirect junk */
543 free(data->req.newurl);
544 data->req.newurl = NULL;
545 free(data->req.location);
546 data->req.location = NULL;
547
548 switch(status) {
549 case CURLE_ABORTED_BY_CALLBACK:
550 case CURLE_READ_ERROR:
551 case CURLE_WRITE_ERROR:
552 /* When we're aborted due to a callback return code it basically have to
553 be counted as premature as there is trouble ahead if we don't. We have
554 many callbacks and protocols work differently, we could potentially do
555 this more fine-grained in the future. */
556 premature = TRUE;
557 default:
558 break;
559 }
560
561 /* this calls the protocol-specific function pointer previously set */
562 if(conn->handler->done)
563 result = conn->handler->done(conn, status, premature);
564 else
565 result = status;
566
567 if(CURLE_ABORTED_BY_CALLBACK != result) {
568 /* avoid this if we already aborted by callback to avoid this calling
569 another callback */
570 CURLcode rc = Curl_pgrsDone(conn);
571 if(!result && rc)
572 result = CURLE_ABORTED_BY_CALLBACK;
573 }
574
575 if(conn->send_pipe.size + conn->recv_pipe.size != 0 &&
576 !data->set.reuse_forbid &&
577 !conn->bits.close) {
578 /* Stop if pipeline is not empty and we do not have to close
579 connection. */
580 data->easy_conn = NULL;
581 DEBUGF(infof(data, "Connection still in use, no more multi_done now!\n"));
582 return CURLE_OK;
583 }
584
585 data->state.done = TRUE; /* called just now! */
586 Curl_resolver_cancel(conn);
587
588 if(conn->dns_entry) {
589 Curl_resolv_unlock(data, conn->dns_entry); /* done with this */
590 conn->dns_entry = NULL;
591 }
592
593 /* if the transfer was completed in a paused state there can be buffered
594 data left to free */
595 for(i = 0; i < data->state.tempcount; i++) {
596 free(data->state.tempwrite[i].buf);
597 }
598 data->state.tempcount = 0;
599
600 /* if data->set.reuse_forbid is TRUE, it means the libcurl client has
601 forced us to close this connection. This is ignored for requests taking
602 place in a NTLM authentication handshake
603
604 if conn->bits.close is TRUE, it means that the connection should be
605 closed in spite of all our efforts to be nice, due to protocol
606 restrictions in our or the server's end
607
608 if premature is TRUE, it means this connection was said to be DONE before
609 the entire request operation is complete and thus we can't know in what
610 state it is for re-using, so we're forced to close it. In a perfect world
611 we can add code that keep track of if we really must close it here or not,
612 but currently we have no such detail knowledge.
613 */
614
615 if((data->set.reuse_forbid
616#if defined(USE_NTLM)
617 && !(conn->ntlm.state == NTLMSTATE_TYPE2 ||
618 conn->proxyntlm.state == NTLMSTATE_TYPE2)
619#endif
620 ) || conn->bits.close || premature) {
621 CURLcode res2 = Curl_disconnect(conn, premature); /* close connection */
622
623 /* If we had an error already, make sure we return that one. But
624 if we got a new error, return that. */
625 if(!result && res2)
626 result = res2;
627 }
628 else {
629 /* the connection is no longer in use */
630 if(ConnectionDone(data, conn)) {
631 /* remember the most recently used connection */
632 data->state.lastconnect = conn;
633
634 infof(data, "Connection #%ld to host %s left intact\n",
635 conn->connection_id,
636 conn->bits.socksproxy ? conn->socks_proxy.host.dispname :
637 conn->bits.httpproxy ? conn->http_proxy.host.dispname :
638 conn->bits.conn_to_host ? conn->conn_to_host.dispname :
639 conn->host.dispname);
640 }
641 else
642 data->state.lastconnect = NULL;
643 }
644
645 *connp = NULL; /* to make the caller of this function better detect that
646 this was either closed or handed over to the connection
647 cache here, and therefore cannot be used from this point on
648 */
649 Curl_free_request_state(data);
650
651 return result;
652}

654CURLMcode curl_multi_remove_handle(struct Curl_multi *multi,
655 struct Curl_easy *data)
656{
657 struct Curl_easy *easy = data;
658 bool premature;
659 bool easy_owns_conn;
660 struct curl_llist_element *e;
661
662 /* First, make some basic checks that the CURLM handle is a good handle */
663 if(!GOOD_MULTI_HANDLE(multi))
664 return CURLM_BAD_HANDLE;
665
666 /* Verify that we got a somewhat good easy handle too */
667 if(!GOOD_EASY_HANDLE(data))
668 return CURLM_BAD_EASY_HANDLE;
669
670 /* Prevent users from trying to remove same easy handle more than once */
671 if(!data->multi)
672 return CURLM_OK; /* it is already removed so let's say it is fine! */
673
674 premature = (data->mstate < CURLM_STATE_COMPLETED) ? TRUE : FALSE;
675 easy_owns_conn = (data->easy_conn && (data->easy_conn->data == easy)) ?
676 TRUE : FALSE;
677
678 /* If the 'state' is not INIT or COMPLETED, we might need to do something
679 nice to put the easy_handle in a good known state when this returns. */
680 if(premature) {
681 /* this handle is "alive" so we need to count down the total number of
682 alive connections when this is removed */
683 multi->num_alive--;
684
685 /* When this handle gets removed, other handles may be able to get the
686 connection */
687 Curl_multi_process_pending_handles(multi);
688 }
689
690 if(data->easy_conn &&
691 data->mstate > CURLM_STATE_DO &&
692 data->mstate < CURLM_STATE_COMPLETED) {
693 /* Set connection owner so that the DONE function closes it. We can
694 safely do this here since connection is killed. */
695 data->easy_conn->data = easy;
696 /* If the handle is in a pipeline and has started sending off its
697 request but not received its response yet, we need to close
698 connection. */
699 streamclose(data->easy_conn, "Removed with partial response");
700 easy_owns_conn = TRUE;
701 }
702
703 /* The timer must be shut down before data->multi is set to NULL,
704 else the timenode will remain in the splay tree after
705 curl_easy_cleanup is called. */
706 Curl_expire_clear(data);
707
708 if(data->easy_conn) {
709
710 /* we must call multi_done() here (if we still own the connection) so that
711 we don't leave a half-baked one around */
712 if(easy_owns_conn) {
713
714 /* multi_done() clears the conn->data field to lose the association
715 between the easy handle and the connection
716
717 Note that this ignores the return code simply because there's
718 nothing really useful to do with it anyway! */
719 (void)multi_done(&data->easy_conn, data->result, premature);
720 }
721 else
722 /* Clear connection pipelines, if multi_done above was not called */
723 Curl_getoff_all_pipelines(data, data->easy_conn);
724 }
725
726 if(data->dns.hostcachetype == HCACHE_MULTI) {
727 /* stop using the multi handle's DNS cache, *after* the possible
728 multi_done() call above */
729 data->dns.hostcache = NULL;
730 data->dns.hostcachetype = HCACHE_NONE;
731 }
732
733 Curl_wildcard_dtor(&data->wildcard);
734
735 /* destroy the timeout list that is held in the easy handle, do this *after*
736 multi_done() as that may actually call Curl_expire that uses this */
737 Curl_llist_destroy(&data->state.timeoutlist, NULL);
738
739 /* as this was using a shared connection cache we clear the pointer to that
740 since we're not part of that multi handle anymore */
741 data->state.conn_cache = NULL;
742
743 /* change state without using multistate(), only to make singlesocket() do
744 what we want */
745 data->mstate = CURLM_STATE_COMPLETED;
746 singlesocket(multi, easy); /* to let the application know what sockets that
747 vanish with this handle */
748
749 /* Remove the association between the connection and the handle */
750 if(data->easy_conn) {
751 data->easy_conn->data = NULL;
752 data->easy_conn = NULL;
753 }
754
755 data->multi = NULL; /* clear the association to this multi handle */
756
757 /* make sure there's no pending message in the queue sent from this easy
758 handle */
759
760 for(e = multi->msglist.head; e; e = e->next) {
761 struct Curl_message *msg = e->ptr;
762
763 if(msg->extmsg.easy_handle == easy) {
764 Curl_llist_remove(&multi->msglist, e, NULL);
765 /* there can only be one from this specific handle */
766 break;
767 }
768 }
769
770 /* make the previous node point to our next */
771 if(data->prev)
772 data->prev->next = data->next;
773 else
774 multi->easyp = data->next; /* point to first node */
775
776 /* make our next point to our previous node */
777 if(data->next)
778 data->next->prev = data->prev;
779 else
780 multi->easylp = data->prev; /* point to last node */
781
782 /* NOTE NOTE NOTE
783 We do not touch the easy handle here! */
784 multi->num_easy--; /* one less to care about now */
785
786 update_timer(multi);
787 return CURLM_OK;
788}

853/* returns bitmapped flags for this handle and its sockets */
854static int multi_getsock(struct Curl_easy *data,
855 curl_socket_t *socks, /* points to numsocks number
856 of sockets */
857 int numsocks)
858{
859 /* If the pipe broke, or if there's no connection left for this easy handle,
860 then we MUST bail out now with no bitmask set. The no connection case can
861 happen when this is called from curl_multi_remove_handle() =>
862 singlesocket() => multi_getsock().
863 */
864 if(data->state.pipe_broke || !data->easy_conn)
865 return 0;
866
867 if(data->mstate > CURLM_STATE_CONNECT &&
868 data->mstate < CURLM_STATE_COMPLETED) {
869 /* Set up ownership correctly */
870 data->easy_conn->data = data;
871 }
872
873 switch(data->mstate) {
874 default:
875#if 0 /* switch back on these cases to get the compiler to check for all enums
876 to be present */
877 case CURLM_STATE_TOOFAST: /* returns 0, so will not select. */
878 case CURLM_STATE_COMPLETED:
879 case CURLM_STATE_MSGSENT:
880 case CURLM_STATE_INIT:
881 case CURLM_STATE_CONNECT:
882 case CURLM_STATE_WAITDO:
883 case CURLM_STATE_DONE:
884 case CURLM_STATE_LAST:
885 /* this will get called with CURLM_STATE_COMPLETED when a handle is
886 removed */
887#endif
888 return 0;
889
890 case CURLM_STATE_WAITRESOLVE:
891 return Curl_resolver_getsock(data->easy_conn, socks, numsocks);
892
893 case CURLM_STATE_PROTOCONNECT:
894 case CURLM_STATE_SENDPROTOCONNECT:
895 return Curl_protocol_getsock(data->easy_conn, socks, numsocks);
896
897 case CURLM_STATE_DO:
898 case CURLM_STATE_DOING:
899 return Curl_doing_getsock(data->easy_conn, socks, numsocks);
900
901 case CURLM_STATE_WAITPROXYCONNECT:
902 return waitproxyconnect_getsock(data->easy_conn, socks, numsocks);
903
904 case CURLM_STATE_WAITCONNECT:
905 return waitconnect_getsock(data->easy_conn, socks, numsocks);
906
907 case CURLM_STATE_DO_MORE:
908 return domore_getsock(data->easy_conn, socks, numsocks);
909
910 case CURLM_STATE_DO_DONE: /* since is set after DO is completed, we switch
911 to waiting for the same as the *PERFORM
912 states */
913 case CURLM_STATE_PERFORM:
914 case CURLM_STATE_WAITPERFORM:
915 return Curl_single_getsock(data->easy_conn, socks, numsocks);
916 }
917
918}

920CURLMcode curl_multi_fdset(struct Curl_multi *multi,
921 fd_set *read_fd_set, fd_set *write_fd_set,
922 fd_set *exc_fd_set, int *max_fd)
923{
924 /* Scan through all the easy handles to get the file descriptors set.
925 Some easy handles may not have connected to the remote host yet,
926 and then we must make sure that is done. */
927 struct Curl_easy *data;
928 int this_max_fd = -1;
929 curl_socket_t sockbunch[MAX_SOCKSPEREASYHANDLE];
930 int bitmap;
931 int i;
932 (void)exc_fd_set; /* not used */
933
934 if(!GOOD_MULTI_HANDLE(multi))
935 return CURLM_BAD_HANDLE;
936
937 data = multi->easyp;
938 while(data) {
939 bitmap = multi_getsock(data, sockbunch, MAX_SOCKSPEREASYHANDLE);
940
941 for(i = 0; i< MAX_SOCKSPEREASYHANDLE; i++) {
942 curl_socket_t s = CURL_SOCKET_BAD;
943
944 if((bitmap & GETSOCK_READSOCK(i)) && VALID_SOCK((sockbunch[i]))) {
945 FD_SET(sockbunch[i], read_fd_set);
946 s = sockbunch[i];
947 }
948 if((bitmap & GETSOCK_WRITESOCK(i)) && VALID_SOCK((sockbunch[i]))) {
949 FD_SET(sockbunch[i], write_fd_set);
950 s = sockbunch[i];
951 }
952 if(s == CURL_SOCKET_BAD)
953 /* this socket is unused, break out of loop */
954 break;
955 if((int)s > this_max_fd)
956 this_max_fd = (int)s;
957 }
958
959 data = data->next; /* check next handle */
960 }
961
962 *max_fd = this_max_fd;
963
964 return CURLM_OK;
965}

968
969CURLMcode curl_multi_wait(struct Curl_multi *multi,
970 struct curl_waitfd extra_fds[],
971 unsigned int extra_nfds,
972 int timeout_ms,
973 int *ret)
974{
975 struct Curl_easy *data;
976 curl_socket_t sockbunch[MAX_SOCKSPEREASYHANDLE];
977 int bitmap;
978 unsigned int i;
979 unsigned int nfds = 0;
980 unsigned int curlfds;
981 struct pollfd *ufds = NULL;
982 bool ufds_malloc = FALSE;
983 long timeout_internal;
984 int retcode = 0;
985 struct pollfd a_few_on_stack[NUM_POLLS_ON_STACK];
986
987 if(!GOOD_MULTI_HANDLE(multi))
988 return CURLM_BAD_HANDLE;
989
990 /* If the internally desired timeout is actually shorter than requested from
991 the outside, then use the shorter time! But only if the internal timer
992 is actually larger than -1! */
993 (void)multi_timeout(multi, &timeout_internal);
994 if((timeout_internal >= 0) && (timeout_internal < (long)timeout_ms))
995 timeout_ms = (int)timeout_internal;
996
997 /* Count up how many fds we have from the multi handle */
998 data = multi->easyp;
999 while(data) {
1000 bitmap = multi_getsock(data, sockbunch, MAX_SOCKSPEREASYHANDLE);
1001
1002 for(i = 0; i< MAX_SOCKSPEREASYHANDLE; i++) {
1003 curl_socket_t s = CURL_SOCKET_BAD;
1004
1005 if(bitmap & GETSOCK_READSOCK(i)) {
1006 ++nfds;
1007 s = sockbunch[i];
1008 }
1009 if(bitmap & GETSOCK_WRITESOCK(i)) {
1010 ++nfds;
1011 s = sockbunch[i];
1012 }
1013 if(s == CURL_SOCKET_BAD) {
1014 break;
1015 }
1016 }
1017
1018 data = data->next; /* check next handle */
1019 }
1020
1021 curlfds = nfds; /* number of internal file descriptors */
1022 nfds += extra_nfds; /* add the externally provided ones */
1023
1024 if(nfds) {
1025 if(nfds > NUM_POLLS_ON_STACK) {
1026 /* 'nfds' is a 32 bit value and 'struct pollfd' is typically 8 bytes
1027 big, so at 2^29 sockets this value might wrap. When a process gets
1028 the capability to actually handle over 500 million sockets this
1029 calculation needs a integer overflow check. */
1030 ufds = malloc(nfds * sizeof(struct pollfd));
1031 if(!ufds)
1032 return CURLM_OUT_OF_MEMORY;
1033 ufds_malloc = TRUE;
1034 }
1035 else
1036 ufds = &a_few_on_stack[0];
1037 }
1038 nfds = 0;
1039
1040 /* only do the second loop if we found descriptors in the first stage run
1041 above */
1042
1043 if(curlfds) {
1044 /* Add the curl handles to our pollfds first */
1045 data = multi->easyp;
1046 while(data) {
1047 bitmap = multi_getsock(data, sockbunch, MAX_SOCKSPEREASYHANDLE);
1048
1049 for(i = 0; i< MAX_SOCKSPEREASYHANDLE; i++) {
1050 curl_socket_t s = CURL_SOCKET_BAD;
1051
1052 if(bitmap & GETSOCK_READSOCK(i)) {
1053 ufds[nfds].fd = sockbunch[i];
1054 ufds[nfds].events = POLLIN;
1055 ++nfds;
1056 s = sockbunch[i];
1057 }
1058 if(bitmap & GETSOCK_WRITESOCK(i)) {
1059 ufds[nfds].fd = sockbunch[i];
1060 ufds[nfds].events = POLLOUT;
1061 ++nfds;
1062 s = sockbunch[i];
1063 }
1064 if(s == CURL_SOCKET_BAD) {
1065 break;
1066 }
1067 }
1068
1069 data = data->next; /* check next handle */
1070 }
1071 }
1072
1073 /* Add external file descriptions from poll-like struct curl_waitfd */
1074 for(i = 0; i < extra_nfds; i++) {
1075 ufds[nfds].fd = extra_fds[i].fd;
1076 ufds[nfds].events = 0;
1077 if(extra_fds[i].events & CURL_WAIT_POLLIN)
1078 ufds[nfds].events |= POLLIN;
1079 if(extra_fds[i].events & CURL_WAIT_POLLPRI)
1080 ufds[nfds].events |= POLLPRI;
1081 if(extra_fds[i].events & CURL_WAIT_POLLOUT)
1082 ufds[nfds].events |= POLLOUT;
1083 ++nfds;
1084 }
1085
1086 if(nfds) {
1087 int pollrc;
1088 /* wait... */
1089 pollrc = Curl_poll(ufds, nfds, timeout_ms);
1090 DEBUGF(infof(data, "Curl_poll(%d ds, %d ms) == %d\n",
1091 nfds, timeout_ms, pollrc));
1092
1093 if(pollrc > 0) {
1094 retcode = pollrc;
1095 /* copy revents results from the poll to the curl_multi_wait poll
1096 struct, the bit values of the actual underlying poll() implementation
1097 may not be the same as the ones in the public libcurl API! */
1098 for(i = 0; i < extra_nfds; i++) {
1099 unsigned short mask = 0;
1100 unsigned r = ufds[curlfds + i].revents;
1101
1102 if(r & POLLIN)
1103 mask |= CURL_WAIT_POLLIN;
1104 if(r & POLLOUT)
1105 mask |= CURL_WAIT_POLLOUT;
1106 if(r & POLLPRI)
1107 mask |= CURL_WAIT_POLLPRI;
1108
1109 extra_fds[i].revents = mask;
1110 }
1111 }
1112 }
1113
1114 if(ufds_malloc)
1115 free(ufds);
1116 if(ret)
1117 *ret = retcode;
1118 return CURLM_OK;
1119}

1171static CURLcode multi_reconnect_request(struct connectdata **connp)
1172{
1173 CURLcode result = CURLE_OK;
1174 struct connectdata *conn = *connp;
1175 struct Curl_easy *data = conn->data;
1176
1177 /* This was a re-use of a connection and we got a write error in the
1178 * DO-phase. Then we DISCONNECT this connection and have another attempt to
1179 * CONNECT and then DO again! The retry cannot possibly find another
1180 * connection to re-use, since we only keep one possible connection for
1181 * each. */
1182
1183 infof(data, "Re-used connection seems dead, get a new one\n");
1184
1185 connclose(conn, "Reconnect dead connection"); /* enforce close */
1186 result = multi_done(&conn, result, FALSE); /* we are so done with this */
1187
1188 /* conn may no longer be a good pointer, clear it to avoid mistakes by
1189 parent functions */
1190 *connp = NULL;
1191
1192 /*
1193 * We need to check for CURLE_SEND_ERROR here as well. This could happen
1194 * when the request failed on a FTP connection and thus multi_done() itself
1195 * tried to use the connection (again).
1196 */
1197 if(!result || (CURLE_SEND_ERROR == result)) {
1198 bool async;
1199 bool protocol_done = TRUE;
1200
1201 /* Now, redo the connect and get a new connection */
1202 result = Curl_connect(data, connp, &async, &protocol_done);
1203 if(!result) {
1204 /* We have connected or sent away a name resolve query fine */
1205
1206 conn = *connp; /* setup conn to again point to something nice */
1207 if(async) {
1208 /* Now, if async is TRUE here, we need to wait for the name
1209 to resolve */
1210 result = Curl_resolver_wait_resolv(conn, NULL);
1211 if(result)
1212 return result;
1213
1214 /* Resolved, continue with the connection */
1215 result = Curl_async_resolved(conn, &protocol_done);
1216 if(result)
1217 return result;
1218 }
1219 }
1220 }
1221
1222 return result;
1223}

1239static CURLcode multi_do(struct connectdata **connp, bool *done)
1240{
1241 CURLcode result = CURLE_OK;
1242 struct connectdata *conn = *connp;
1243 struct Curl_easy *data = conn->data;
1244
1245 if(conn->handler->do_it) {
1246 /* generic protocol-specific function pointer set in curl_connect() */
1247 result = conn->handler->do_it(conn, done);
1248
1249 /* This was formerly done in transfer.c, but we better do it here */
1250 if((CURLE_SEND_ERROR == result) && conn->bits.reuse) {
1251 /*
1252 * If the connection is using an easy handle, call reconnect
1253 * to re-establish the connection. Otherwise, let the multi logic
1254 * figure out how to re-establish the connection.
1255 */
1256 if(!data->multi) {
1257 result = multi_reconnect_request(connp);
1258
1259 if(!result) {
1260 /* ... finally back to actually retry the DO phase */
1261 conn = *connp; /* re-assign conn since multi_reconnect_request
1262 creates a new connection */
1263 result = conn->handler->do_it(conn, done);
1264 }
1265 }
1266 else
1267 return result;
1268 }
1269
1270 if(!result && *done)
1271 /* do_complete must be called after the protocol-specific DO function */
1272 do_complete(conn);
1273 }
1274 return result;
1275}

1279 * second stage DO state which (wrongly) was introduced to support FTP's
1280 * second connection.
1281 *
1282 * TODO: A future libcurl should be able to work away this state.
1283 *
1284 * 'complete' can return 0 for incomplete, 1 for done and -1 for go back to

1298 /* do_complete must be called after the protocol-specific DO function */
1299 do_complete(conn);
1300
1301 return result;

1304static CURLMcode multi_runsingle(struct Curl_multi *multi,
1305 struct curltime now,
1306 struct Curl_easy *data)
1307{
1308 struct Curl_message *msg = NULL;
1309 bool connected;
1310 bool async;
1311 bool protocol_connect = FALSE;
1312 bool dophase_done = FALSE;
1313 bool done = FALSE;
1314 CURLMcode rc;
1315 CURLcode result = CURLE_OK;
1316 struct SingleRequest *k;
1317 time_t timeout_ms;
1318 time_t recv_timeout_ms;
1319 time_t send_timeout_ms;
1320 int control;
1321
1322 if(!GOOD_EASY_HANDLE(data))
1323 return CURLM_BAD_EASY_HANDLE;
1324
1325 do {
1326 /* A "stream" here is a logical stream if the protocol can handle that
1327 (HTTP/2), or the full connection for older protocols */
1328 bool stream_error = FALSE;
1329 rc = CURLM_OK;
1330
1331 /* Handle the case when the pipe breaks, i.e., the connection
1332 we're using gets cleaned up and we're left with nothing. */
1333 if(data->state.pipe_broke) {
1334 infof(data, "Pipe broke: handle %p, url = %s\n",
1335 (void *)data, data->state.path);
1336
1337 if(data->mstate < CURLM_STATE_COMPLETED) {
1338 /* Head back to the CONNECT state */
1339 multistate(data, CURLM_STATE_CONNECT);
1340 rc = CURLM_CALL_MULTI_PERFORM;
1341 result = CURLE_OK;
1342 }
1343
1344 data->state.pipe_broke = FALSE;
1345 data->easy_conn = NULL;
1346 continue;
1347 }
1348
1349 if(!data->easy_conn &&
1350 data->mstate > CURLM_STATE_CONNECT &&
1351 data->mstate < CURLM_STATE_DONE) {
1352 /* In all these states, the code will blindly access 'data->easy_conn'
1353 so this is precaution that it isn't NULL. And it silences static
1354 analyzers. */
1355 failf(data, "In state %d with no easy_conn, bail out!\n", data->mstate);
1356 return CURLM_INTERNAL_ERROR;
1357 }
1358
1359 if(multi_ischanged(multi, TRUE)) {
1360 DEBUGF(infof(data, "multi changed, check CONNECT_PEND queue!\n"));
1361 Curl_multi_process_pending_handles(multi);
1362 }
1363
1364 if(data->easy_conn && data->mstate > CURLM_STATE_CONNECT &&
1365 data->mstate < CURLM_STATE_COMPLETED)
1366 /* Make sure we set the connection's current owner */
1367 data->easy_conn->data = data;
1368
1369 if(data->easy_conn &&
1370 (data->mstate >= CURLM_STATE_CONNECT) &&
1371 (data->mstate < CURLM_STATE_COMPLETED)) {
1372 /* we need to wait for the connect state as only then is the start time
1373 stored, but we must not check already completed handles */
1374
1375 timeout_ms = Curl_timeleft(data, &now,
1376 (data->mstate <= CURLM_STATE_WAITDO)?
1377 TRUE:FALSE);
1378
1379 if(timeout_ms < 0) {
1380 /* Handle timed out */
1381 if(data->mstate == CURLM_STATE_WAITRESOLVE)
1382 failf(data, "Resolving timed out after %ld milliseconds",
1383 Curl_tvdiff(now, data->progress.t_startsingle));
1384 else if(data->mstate == CURLM_STATE_WAITCONNECT)
1385 failf(data, "Connection timed out after %ld milliseconds",
1386 Curl_tvdiff(now, data->progress.t_startsingle));
1387 else {
1388 k = &data->req;
1389 if(k->size != -1) {
1390 failf(data, "Operation timed out after %ld milliseconds with %"
1391 CURL_FORMAT_CURL_OFF_T " out of %"
1392 CURL_FORMAT_CURL_OFF_T " bytes received",
1393 Curl_tvdiff(now, data->progress.t_startsingle),
1394 k->bytecount, k->size);
1395 }
1396 else {
1397 failf(data, "Operation timed out after %ld milliseconds with %"
1398 CURL_FORMAT_CURL_OFF_T " bytes received",
1399 Curl_tvdiff(now, data->progress.t_startsingle),
1400 k->bytecount);
1401 }
1402 }
1403
1404 /* Force connection closed if the connection has indeed been used */
1405 if(data->mstate > CURLM_STATE_DO) {
1406 streamclose(data->easy_conn, "Disconnected with pending data");
1407 stream_error = TRUE;
1408 }
1409 result = CURLE_OPERATION_TIMEDOUT;
1410 (void)multi_done(&data->easy_conn, result, TRUE);
1411 /* Skip the statemachine and go directly to error handling section. */
1412 goto statemachine_end;
1413 }
1414 }
1415
1416 switch(data->mstate) {
1417 case CURLM_STATE_INIT:
1418 /* init this transfer. */
1419 result = Curl_pretransfer(data);
1420
1421 if(!result) {
1422 /* after init, go CONNECT */
1423 multistate(data, CURLM_STATE_CONNECT);
1424 Curl_pgrsTime(data, TIMER_STARTOP);
1425 rc = CURLM_CALL_MULTI_PERFORM;
1426 }
1427 break;
1428
1429 case CURLM_STATE_CONNECT_PEND:
1430 /* We will stay here until there is a connection available. Then
1431 we try again in the CURLM_STATE_CONNECT state. */
1432 break;
1433
1434 case CURLM_STATE_CONNECT:
1435 /* Connect. We want to get a connection identifier filled in. */
1436 Curl_pgrsTime(data, TIMER_STARTSINGLE);
1437 result = Curl_connect(data, &data->easy_conn,
1438 &async, &protocol_connect);
1439 if(CURLE_NO_CONNECTION_AVAILABLE == result) {
1440 /* There was no connection available. We will go to the pending
1441 state and wait for an available connection. */
1442 multistate(data, CURLM_STATE_CONNECT_PEND);
1443
1444 /* add this handle to the list of connect-pending handles */
1445 Curl_llist_insert_next(&multi->pending, multi->pending.tail, data,
1446 &data->connect_queue);
1447 result = CURLE_OK;
1448 break;
1449 }
1450
1451 if(!result) {
1452 /* Add this handle to the send or pend pipeline */
1453 result = Curl_add_handle_to_pipeline(data, data->easy_conn);
1454 if(result)
1455 stream_error = TRUE;
1456 else {
1457 if(async)
1458 /* We're now waiting for an asynchronous name lookup */
1459 multistate(data, CURLM_STATE_WAITRESOLVE);
1460 else {
1461 /* after the connect has been sent off, go WAITCONNECT unless the
1462 protocol connect is already done and we can go directly to
1463 WAITDO or DO! */
1464 rc = CURLM_CALL_MULTI_PERFORM;
1465
1466 if(protocol_connect)
1467 multistate(data, Curl_pipeline_wanted(multi, CURLPIPE_HTTP1)?
1468 CURLM_STATE_WAITDO:CURLM_STATE_DO);
1469 else {
1470#ifndef CURL_DISABLE_HTTP
1471 if(Curl_connect_ongoing(data->easy_conn))
1472 multistate(data, CURLM_STATE_WAITPROXYCONNECT);
1473 else
1474#endif
1475 multistate(data, CURLM_STATE_WAITCONNECT);
1476 }
1477 }
1478 }
1479 }
1480 break;
1481
1482 case CURLM_STATE_WAITRESOLVE:
1483 /* awaiting an asynch name resolve to complete */
1484 {
1485 struct Curl_dns_entry *dns = NULL;
1486 struct connectdata *conn = data->easy_conn;
1487 const char *hostname;
1488
1489 if(conn->bits.httpproxy)
1490 hostname = conn->http_proxy.host.name;
1491 else if(conn->bits.conn_to_host)
1492 hostname = conn->conn_to_host.name;
1493 else
1494 hostname = conn->host.name;
1495
1496 /* check if we have the name resolved by now */
1497 dns = Curl_fetch_addr(conn, hostname, (int)conn->port);
1498
1499 if(dns) {
1500#ifdef CURLRES_ASYNCH
1501 conn->async.dns = dns;
1502 conn->async.done = TRUE;
1503#endif
1504 result = CURLE_OK;
1505 infof(data, "Hostname '%s' was found in DNS cache\n", hostname);
1506 }
1507
1508 if(!dns)
1509 result = Curl_resolver_is_resolved(data->easy_conn, &dns);
1510
1511 /* Update sockets here, because the socket(s) may have been
1512 closed and the application thus needs to be told, even if it
1513 is likely that the same socket(s) will again be used further
1514 down. If the name has not yet been resolved, it is likely
1515 that new sockets have been opened in an attempt to contact
1516 another resolver. */
1517 singlesocket(multi, data);
1518
1519 if(dns) {
1520 /* Perform the next step in the connection phase, and then move on
1521 to the WAITCONNECT state */
1522 result = Curl_async_resolved(data->easy_conn, &protocol_connect);
1523
1524 if(result)
1525 /* if Curl_async_resolved() returns failure, the connection struct
1526 is already freed and gone */
1527 data->easy_conn = NULL; /* no more connection */
1528 else {
1529 /* call again please so that we get the next socket setup */
1530 rc = CURLM_CALL_MULTI_PERFORM;
1531 if(protocol_connect)
1532 multistate(data, Curl_pipeline_wanted(multi, CURLPIPE_HTTP1)?
1533 CURLM_STATE_WAITDO:CURLM_STATE_DO);
1534 else {
1535#ifndef CURL_DISABLE_HTTP
1536 if(Curl_connect_ongoing(data->easy_conn))
1537 multistate(data, CURLM_STATE_WAITPROXYCONNECT);
1538 else
1539#endif
1540 multistate(data, CURLM_STATE_WAITCONNECT);
1541 }
1542 }
1543 }
1544
1545 if(result) {
1546 /* failure detected */
1547 stream_error = TRUE;
1548 break;
1549 }
1550 }
1551 break;
1552
1553#ifndef CURL_DISABLE_HTTP
1554 case CURLM_STATE_WAITPROXYCONNECT:
1555 /* this is HTTP-specific, but sending CONNECT to a proxy is HTTP... */
1556 result = Curl_http_connect(data->easy_conn, &protocol_connect);
1557
1558 if(data->easy_conn->bits.proxy_connect_closed) {
1559 rc = CURLM_CALL_MULTI_PERFORM;
1560 /* connect back to proxy again */
1561 result = CURLE_OK;
1562 multi_done(&data->easy_conn, CURLE_OK, FALSE);
1563 multistate(data, CURLM_STATE_CONNECT);
1564 }
1565 else if(!result) {
1566 if((data->easy_conn->http_proxy.proxytype != CURLPROXY_HTTPS ||
1567 data->easy_conn->bits.proxy_ssl_connected[FIRSTSOCKET]) &&
1568 Curl_connect_complete(data->easy_conn)) {
1569 rc = CURLM_CALL_MULTI_PERFORM;
1570 /* initiate protocol connect phase */
1571 multistate(data, CURLM_STATE_SENDPROTOCONNECT);
1572 }
1573 }
1574 break;
1575#endif
1576
1577 case CURLM_STATE_WAITCONNECT:
1578 /* awaiting a completion of an asynch TCP connect */
1579 result = Curl_is_connected(data->easy_conn, FIRSTSOCKET, &connected);
1580 if(connected && !result) {
1581#ifndef CURL_DISABLE_HTTP
1582 if((data->easy_conn->http_proxy.proxytype == CURLPROXY_HTTPS &&
1583 !data->easy_conn->bits.proxy_ssl_connected[FIRSTSOCKET]) ||
1584 Curl_connect_ongoing(data->easy_conn)) {
1585 multistate(data, CURLM_STATE_WAITPROXYCONNECT);
1586 break;
1587 }
1588#endif
1589 rc = CURLM_CALL_MULTI_PERFORM;
1590 multistate(data, data->easy_conn->bits.tunnel_proxy?
1591 CURLM_STATE_WAITPROXYCONNECT:
1592 CURLM_STATE_SENDPROTOCONNECT);
1593 }
1594 else if(result) {
1595 /* failure detected */
1596 /* Just break, the cleaning up is handled all in one place */
1597 stream_error = TRUE;
1598 break;
1599 }
1600 break;
1601
1602 case CURLM_STATE_SENDPROTOCONNECT:
1603 result = Curl_protocol_connect(data->easy_conn, &protocol_connect);
1604 if(!protocol_connect)
1605 /* switch to waiting state */
1606 multistate(data, CURLM_STATE_PROTOCONNECT);
1607 else if(!result) {
1608 /* protocol connect has completed, go WAITDO or DO */
1609 multistate(data, Curl_pipeline_wanted(multi, CURLPIPE_HTTP1)?
1610 CURLM_STATE_WAITDO:CURLM_STATE_DO);
1611 rc = CURLM_CALL_MULTI_PERFORM;
1612 }
1613 else if(result) {
1614 /* failure detected */
1615 Curl_posttransfer(data);
1616 multi_done(&data->easy_conn, result, TRUE);
1617 stream_error = TRUE;
1618 }
1619 break;
1620
1621 case CURLM_STATE_PROTOCONNECT:
1622 /* protocol-specific connect phase */
1623 result = Curl_protocol_connecting(data->easy_conn, &protocol_connect);
1624 if(!result && protocol_connect) {
1625 /* after the connect has completed, go WAITDO or DO */
1626 multistate(data, Curl_pipeline_wanted(multi, CURLPIPE_HTTP1)?
1627 CURLM_STATE_WAITDO:CURLM_STATE_DO);
1628 rc = CURLM_CALL_MULTI_PERFORM;
1629 }
1630 else if(result) {
1631 /* failure detected */
1632 Curl_posttransfer(data);
1633 multi_done(&data->easy_conn, result, TRUE);
1634 stream_error = TRUE;
1635 }
1636 break;
1637
1638 case CURLM_STATE_WAITDO:
1639 /* Wait for our turn to DO when we're pipelining requests */
1640 if(Curl_pipeline_checkget_write(data, data->easy_conn)) {
1641 /* Grabbed the channel */
1642 multistate(data, CURLM_STATE_DO);
1643 rc = CURLM_CALL_MULTI_PERFORM;
1644 }
1645 break;
1646
1647 case CURLM_STATE_DO:
1648 if(data->set.connect_only) {
1649 /* keep connection open for application to use the socket */
1650 connkeep(data->easy_conn, "CONNECT_ONLY");
1651 multistate(data, CURLM_STATE_DONE);
1652 result = CURLE_OK;
1653 rc = CURLM_CALL_MULTI_PERFORM;
1654 }
1655 else {
1656 /* Perform the protocol's DO action */
1657 result = multi_do(&data->easy_conn, &dophase_done);
1658
1659 /* When multi_do() returns failure, data->easy_conn might be NULL! */
1660
1661 if(!result) {
1662 if(!dophase_done) {
1663 /* some steps needed for wildcard matching */
1664 if(data->set.wildcardmatch) {
1665 struct WildcardData *wc = &data->wildcard;
1666 if(wc->state == CURLWC_DONE || wc->state == CURLWC_SKIP) {
1667 /* skip some states if it is important */
1668 multi_done(&data->easy_conn, CURLE_OK, FALSE);
1669 multistate(data, CURLM_STATE_DONE);
1670 rc = CURLM_CALL_MULTI_PERFORM;
1671 break;
1672 }
1673 }
1674 /* DO was not completed in one function call, we must continue
1675 DOING... */
1676 multistate(data, CURLM_STATE_DOING);
1677 rc = CURLM_OK;
1678 }
1679
1680 /* after DO, go DO_DONE... or DO_MORE */
1681 else if(data->easy_conn->bits.do_more) {
1682 /* we're supposed to do more, but we need to sit down, relax
1683 and wait a little while first */
1684 multistate(data, CURLM_STATE_DO_MORE);
1685 rc = CURLM_OK;
1686 }
1687 else {
1688 /* we're done with the DO, now DO_DONE */
1689 multistate(data, CURLM_STATE_DO_DONE);
1690 rc = CURLM_CALL_MULTI_PERFORM;
1691 }
1692 }
1693 else if((CURLE_SEND_ERROR == result) &&
1694 data->easy_conn->bits.reuse) {
1695 /*
1696 * In this situation, a connection that we were trying to use
1697 * may have unexpectedly died. If possible, send the connection
1698 * back to the CONNECT phase so we can try again.
1699 */
1700 char *newurl = NULL;
1701 followtype follow = FOLLOW_NONE;
1702 CURLcode drc;
1703 bool retry = FALSE;
1704
1705 drc = Curl_retry_request(data->easy_conn, &newurl);
1706 if(drc) {
1707 /* a failure here pretty much implies an out of memory */
1708 result = drc;
1709 stream_error = TRUE;
1710 }
1711 else
1712 retry = (newurl)?TRUE:FALSE;
1713
1714 Curl_posttransfer(data);
1715 drc = multi_done(&data->easy_conn, result, FALSE);
1716
1717 /* When set to retry the connection, we must to go back to
1718 * the CONNECT state */
1719 if(retry) {
1720 if(!drc || (drc == CURLE_SEND_ERROR)) {
1721 follow = FOLLOW_RETRY;
1722 drc = Curl_follow(data, newurl, follow);
1723 if(!drc) {
1724 multistate(data, CURLM_STATE_CONNECT);
1725 rc = CURLM_CALL_MULTI_PERFORM;
1726 result = CURLE_OK;
1727 }
1728 else {
1729 /* Follow failed */
1730 result = drc;
1731 }
1732 }
1733 else {
1734 /* done didn't return OK or SEND_ERROR */
1735 result = drc;
1736 }
1737 }
1738 else {
1739 /* Have error handler disconnect conn if we can't retry */
1740 stream_error = TRUE;
1741 }
1742 free(newurl);
1743 }
1744 else {
1745 /* failure detected */
1746 Curl_posttransfer(data);
1747 if(data->easy_conn)
1748 multi_done(&data->easy_conn, result, FALSE);
1749 stream_error = TRUE;
1750 }
1751 }
1752 break;
1753
1754 case CURLM_STATE_DOING:
1755 /* we continue DOING until the DO phase is complete */
1756 result = Curl_protocol_doing(data->easy_conn,
1757 &dophase_done);
1758 if(!result) {
1759 if(dophase_done) {
1760 /* after DO, go DO_DONE or DO_MORE */
1761 multistate(data, data->easy_conn->bits.do_more?
1762 CURLM_STATE_DO_MORE:
1763 CURLM_STATE_DO_DONE);
1764 rc = CURLM_CALL_MULTI_PERFORM;
1765 } /* dophase_done */
1766 }
1767 else {
1768 /* failure detected */
1769 Curl_posttransfer(data);
1770 multi_done(&data->easy_conn, result, FALSE);
1771 stream_error = TRUE;
1772 }
1773 break;
1774
1775 case CURLM_STATE_DO_MORE:
1776 /*
1777 * When we are connected, DO MORE and then go DO_DONE
1778 */
1779 result = multi_do_more(data->easy_conn, &control);
1780
1781 /* No need to remove this handle from the send pipeline here since that
1782 is done in multi_done() */
1783 if(!result) {
1784 if(control) {
1785 /* if positive, advance to DO_DONE
1786 if negative, go back to DOING */
1787 multistate(data, control == 1?
1788 CURLM_STATE_DO_DONE:
1789 CURLM_STATE_DOING);
1790 rc = CURLM_CALL_MULTI_PERFORM;
1791 }
1792 else
1793 /* stay in DO_MORE */
1794 rc = CURLM_OK;
1795 }
1796 else {
1797 /* failure detected */
1798 Curl_posttransfer(data);
1799 multi_done(&data->easy_conn, result, FALSE);
1800 stream_error = TRUE;
1801 }
1802 break;
1803
1804 case CURLM_STATE_DO_DONE:
1805 /* Move ourselves from the send to recv pipeline */
1806 Curl_move_handle_from_send_to_recv_pipe(data, data->easy_conn);
1807 /* Check if we can move pending requests to send pipe */
1808 Curl_multi_process_pending_handles(multi);
1809
1810 /* Only perform the transfer if there's a good socket to work with.
1811 Having both BAD is a signal to skip immediately to DONE */
1812 if((data->easy_conn->sockfd != CURL_SOCKET_BAD) ||
1813 (data->easy_conn->writesockfd != CURL_SOCKET_BAD))
1814 multistate(data, CURLM_STATE_WAITPERFORM);
1815 else
1816 multistate(data, CURLM_STATE_DONE);
1817 rc = CURLM_CALL_MULTI_PERFORM;
1818 break;
1819
1820 case CURLM_STATE_WAITPERFORM:
1821 /* Wait for our turn to PERFORM */
1822 if(Curl_pipeline_checkget_read(data, data->easy_conn)) {
1823 /* Grabbed the channel */
1824 multistate(data, CURLM_STATE_PERFORM);
1825 rc = CURLM_CALL_MULTI_PERFORM;
1826 }
1827 break;
1828
1829 case CURLM_STATE_TOOFAST: /* limit-rate exceeded in either direction */
1830 /* if both rates are within spec, resume transfer */
1831 if(Curl_pgrsUpdate(data->easy_conn))
1832 result = CURLE_ABORTED_BY_CALLBACK;
1833 else
1834 result = Curl_speedcheck(data, now);
1835
1836 if(!result) {
1837 send_timeout_ms = 0;
1838 if(data->set.max_send_speed > 0)
1839 send_timeout_ms = Curl_pgrsLimitWaitTime(data->progress.uploaded,
1840 data->progress.ul_limit_size,
1841 data->set.max_send_speed,
1842 data->progress.ul_limit_start,
1843 now);
1844
1845 recv_timeout_ms = 0;
1846 if(data->set.max_recv_speed > 0)
1847 recv_timeout_ms = Curl_pgrsLimitWaitTime(data->progress.downloaded,
1848 data->progress.dl_limit_size,
1849 data->set.max_recv_speed,
1850 data->progress.dl_limit_start,
1851 now);
1852
1853 if(send_timeout_ms <= 0 && recv_timeout_ms <= 0)
1854 multistate(data, CURLM_STATE_PERFORM);
1855 else if(send_timeout_ms >= recv_timeout_ms)
1856 Curl_expire(data, send_timeout_ms, EXPIRE_TOOFAST);
1857 else
1858 Curl_expire(data, recv_timeout_ms, EXPIRE_TOOFAST);
1859 }
1860 break;
1861
1862 case CURLM_STATE_PERFORM:
1863 {
1864 char *newurl = NULL;
1865 bool retry = FALSE;
1866 bool comeback = FALSE;
1867
1868 /* check if over send speed */
1869 send_timeout_ms = 0;
1870 if(data->set.max_send_speed > 0)
1871 send_timeout_ms = Curl_pgrsLimitWaitTime(data->progress.uploaded,
1872 data->progress.ul_limit_size,
1873 data->set.max_send_speed,
1874 data->progress.ul_limit_start,
1875 now);
1876
1877 /* check if over recv speed */
1878 recv_timeout_ms = 0;
1879 if(data->set.max_recv_speed > 0)
1880 recv_timeout_ms = Curl_pgrsLimitWaitTime(data->progress.downloaded,
1881 data->progress.dl_limit_size,
1882 data->set.max_recv_speed,
1883 data->progress.dl_limit_start,
1884 now);
1885
1886 if(send_timeout_ms > 0 || recv_timeout_ms > 0) {
1887 multistate(data, CURLM_STATE_TOOFAST);
1888 if(send_timeout_ms >= recv_timeout_ms)
1889 Curl_expire(data, send_timeout_ms, EXPIRE_TOOFAST);
1890 else
1891 Curl_expire(data, recv_timeout_ms, EXPIRE_TOOFAST);
1892 break;
1893 }
1894
1895 /* read/write data if it is ready to do so */
1896 result = Curl_readwrite(data->easy_conn, data, &done, &comeback);
1897
1898 k = &data->req;
1899
1900 if(!(k->keepon & KEEP_RECV))
1901 /* We're done receiving */
1902 Curl_pipeline_leave_read(data->easy_conn);
1903
1904 if(!(k->keepon & KEEP_SEND))
1905 /* We're done sending */
1906 Curl_pipeline_leave_write(data->easy_conn);
1907
1908 if(done || (result == CURLE_RECV_ERROR)) {
1909 /* If CURLE_RECV_ERROR happens early enough, we assume it was a race
1910 * condition and the server closed the re-used connection exactly when
1911 * we wanted to use it, so figure out if that is indeed the case.
1912 */
1913 CURLcode ret = Curl_retry_request(data->easy_conn, &newurl);
1914 if(!ret)
1915 retry = (newurl)?TRUE:FALSE;
1916
1917 if(retry) {
1918 /* if we are to retry, set the result to OK and consider the
1919 request as done */
1920 result = CURLE_OK;
1921 done = TRUE;
1922 }
1923 }
1924
1925 if(result) {
1926 /*
1927 * The transfer phase returned error, we mark the connection to get
1928 * closed to prevent being re-used. This is because we can't possibly
1929 * know if the connection is in a good shape or not now. Unless it is
1930 * a protocol which uses two "channels" like FTP, as then the error
1931 * happened in the data connection.
1932 */
1933
1934 if(!(data->easy_conn->handler->flags & PROTOPT_DUAL) &&
1935 result != CURLE_HTTP2_STREAM)
1936 streamclose(data->easy_conn, "Transfer returned error");
1937
1938 Curl_posttransfer(data);
1939 multi_done(&data->easy_conn, result, TRUE);
1940 }
1941 else if(done) {
1942 followtype follow = FOLLOW_NONE;
1943
1944 /* call this even if the readwrite function returned error */
1945 Curl_posttransfer(data);
1946
1947 /* we're no longer receiving */
1948 Curl_removeHandleFromPipeline(data, &data->easy_conn->recv_pipe);
1949
1950 /* expire the new receiving pipeline head */
1951 if(data->easy_conn->recv_pipe.head)
1952 Curl_expire(data->easy_conn->recv_pipe.head->ptr, 0, EXPIRE_RUN_NOW);
1953
1954 /* Check if we can move pending requests to send pipe */
1955 Curl_multi_process_pending_handles(multi);
1956
1957 /* When we follow redirects or is set to retry the connection, we must
1958 to go back to the CONNECT state */
1959 if(data->req.newurl || retry) {
1960 if(!retry) {
1961 /* if the URL is a follow-location and not just a retried request
1962 then figure out the URL here */
1963 free(newurl);
1964 newurl = data->req.newurl;
1965 data->req.newurl = NULL;
1966 follow = FOLLOW_REDIR;
1967 }
1968 else
1969 follow = FOLLOW_RETRY;
1970 result = multi_done(&data->easy_conn, CURLE_OK, FALSE);
1971 if(!result) {
1972 result = Curl_follow(data, newurl, follow);
1973 if(!result) {
1974 multistate(data, CURLM_STATE_CONNECT);
1975 rc = CURLM_CALL_MULTI_PERFORM;
1976 }
1977 }
1978 }
1979 else {
1980 /* after the transfer is done, go DONE */
1981
1982 /* but first check to see if we got a location info even though we're
1983 not following redirects */
1984 if(data->req.location) {
1985 free(newurl);
1986 newurl = data->req.location;
1987 data->req.location = NULL;
1988 result = Curl_follow(data, newurl, FOLLOW_FAKE);
1989 if(result)
1990 stream_error = TRUE;
1991 }
1992
1993 multistate(data, CURLM_STATE_DONE);
1994 rc = CURLM_CALL_MULTI_PERFORM;
1995 }
1996 }
1997 else if(comeback)
1998 rc = CURLM_CALL_MULTI_PERFORM;
1999
2000 free(newurl);
2001 break;
2002 }
2003
2004 case CURLM_STATE_DONE:
2005 /* this state is highly transient, so run another loop after this */
2006 rc = CURLM_CALL_MULTI_PERFORM;
2007
2008 if(data->easy_conn) {
2009 CURLcode res;
2010
2011 /* Remove ourselves from the receive pipeline, if we are there. */
2012 Curl_removeHandleFromPipeline(data, &data->easy_conn->recv_pipe);
2013 /* Check if we can move pending requests to send pipe */
2014 Curl_multi_process_pending_handles(multi);
2015
2016 /* post-transfer command */
2017 res = multi_done(&data->easy_conn, result, FALSE);
2018
2019 /* allow a previously set error code take precedence */
2020 if(!result)
2021 result = res;
2022
2023 /*
2024 * If there are other handles on the pipeline, multi_done won't set
2025 * easy_conn to NULL. In such a case, curl_multi_remove_handle() can
2026 * access free'd data, if the connection is free'd and the handle
2027 * removed before we perform the processing in CURLM_STATE_COMPLETED
2028 */
2029 if(data->easy_conn)
2030 data->easy_conn = NULL;
2031 }
2032
2033 if(data->set.wildcardmatch) {
2034 if(data->wildcard.state != CURLWC_DONE) {
2035 /* if a wildcard is set and we are not ending -> lets start again
2036 with CURLM_STATE_INIT */
2037 multistate(data, CURLM_STATE_INIT);
2038 break;
2039 }
2040 }
2041
2042 /* after we have DONE what we're supposed to do, go COMPLETED, and
2043 it doesn't matter what the multi_done() returned! */
2044 multistate(data, CURLM_STATE_COMPLETED);
2045 break;
2046
2047 case CURLM_STATE_COMPLETED:
2048 /* this is a completed transfer, it is likely to still be connected */
2049
2050 /* This node should be delinked from the list now and we should post
2051 an information message that we are complete. */
2052
2053 /* Important: reset the conn pointer so that we don't point to memory
2054 that could be freed anytime */
2055 data->easy_conn = NULL;
2056
2057 Curl_expire_clear(data); /* stop all timers */
2058 break;
2059
2060 case CURLM_STATE_MSGSENT:
2061 data->result = result;
2062 return CURLM_OK; /* do nothing */
2063
2064 default:
2065 return CURLM_INTERNAL_ERROR;
2066 }
2067 statemachine_end:
2068
2069 if(data->mstate < CURLM_STATE_COMPLETED) {
2070 if(result) {
2071 /*
2072 * If an error was returned, and we aren't in completed state now,
2073 * then we go to completed and consider this transfer aborted.
2074 */
2075
2076 /* NOTE: no attempt to disconnect connections must be made
2077 in the case blocks above - cleanup happens only here */
2078
2079 data->state.pipe_broke = FALSE;
2080
2081 /* Check if we can move pending requests to send pipe */
2082 Curl_multi_process_pending_handles(multi);
2083
2084 if(data->easy_conn) {
2085 /* if this has a connection, unsubscribe from the pipelines */
2086 Curl_pipeline_leave_write(data->easy_conn);
2087 Curl_pipeline_leave_read(data->easy_conn);

2318/*
2319 * singlesocket() checks what sockets we deal with and their "action state"
2320 * and if we have a different state in any of those sockets from last time we
2321 * call the callback accordingly.
2322 */
2323static void singlesocket(struct Curl_multi *multi,
2324 struct Curl_easy *data)
2325{
2326 curl_socket_t socks[MAX_SOCKSPEREASYHANDLE];
2327 int i;
2328 struct Curl_sh_entry *entry;
2329 curl_socket_t s;
2330 int num;
2331 unsigned int curraction;
2332
2333 for(i = 0; i< MAX_SOCKSPEREASYHANDLE; i++)
2334 socks[i] = CURL_SOCKET_BAD;
2335
2336 /* Fill in the 'current' struct with the state as it is now: what sockets to
2337 supervise and for what actions */
2338 curraction = multi_getsock(data, socks, MAX_SOCKSPEREASYHANDLE);
2339
2340 /* We have 0 .. N sockets already and we get to know about the 0 .. M
2341 sockets we should have from now on. Detect the differences, remove no
2342 longer supervised ones and add new ones */
2343
2344 /* walk over the sockets we got right now */
2345 for(i = 0; (i< MAX_SOCKSPEREASYHANDLE) &&
2346 (curraction & (GETSOCK_READSOCK(i) | GETSOCK_WRITESOCK(i)));
2347 i++) {
2348 int action = CURL_POLL_NONE;
2349
2350 s = socks[i];
2351
2352 /* get it from the hash */
2353 entry = sh_getentry(&multi->sockhash, s);
2354
2355 if(curraction & GETSOCK_READSOCK(i))
2356 action |= CURL_POLL_IN;
2357 if(curraction & GETSOCK_WRITESOCK(i))
2358 action |= CURL_POLL_OUT;
2359
2360 if(entry) {
2361 /* yeps, already present so check if it has the same action set */
2362 if(entry->action == action)
2363 /* same, continue */
2364 continue;
2365 }
2366 else {
2367 /* this is a socket we didn't have before, add it! */
2368 entry = sh_addentry(&multi->sockhash, s, data);
2369 if(!entry)
2370 /* fatal */
2371 return;
2372 }
2373
2374 /* we know (entry != NULL) at this point, see the logic above */
2375 if(multi->socket_cb)
2376 multi->socket_cb(data,
2377 s,
2378 action,
2379 multi->socket_userp,
2380 entry->socketp);
2381
2382 entry->action = action; /* store the current action state */
2383 }
2384
2385 num = i; /* number of sockets */
2386
2387 /* when we've walked over all the sockets we should have right now, we must
2388 make sure to detect sockets that are removed */
2389 for(i = 0; i< data->numsocks; i++) {
2390 int j;
2391 s = data->sockets[i];
2392 for(j = 0; j<num; j++) {
2393 if(s == socks[j]) {
2394 /* this is still supervised */
2395 s = CURL_SOCKET_BAD;
2396 break;
2397 }
2398 }
2399
2400 entry = sh_getentry(&multi->sockhash, s);
2401 if(entry) {
2402 /* this socket has been removed. Tell the app to remove it */
2403 bool remove_sock_from_hash = TRUE;
2404
2405 /* check if the socket to be removed serves a connection which has
2406 other easy-s in a pipeline. In this case the socket should not be
2407 removed. */
2408 struct connectdata *easy_conn = data->easy_conn;
2409 if(easy_conn) {
2410 if(easy_conn->recv_pipe.size > 1) {
2411 /* the handle should not be removed from the pipe yet */
2412 remove_sock_from_hash = FALSE;
2413
2414 /* Update the sockhash entry to instead point to the next in line
2415 for the recv_pipe, or the first (in case this particular easy
2416 isn't already) */
2417 if(entry->easy == data) {
2418 if(Curl_recvpipe_head(data, easy_conn))
2419 entry->easy = easy_conn->recv_pipe.head->next->ptr;
2420 else
2421 entry->easy = easy_conn->recv_pipe.head->ptr;
2422 }
2423 }
2424 if(easy_conn->send_pipe.size > 1) {
2425 /* the handle should not be removed from the pipe yet */
2426 remove_sock_from_hash = FALSE;
2427
2428 /* Update the sockhash entry to instead point to the next in line
2429 for the send_pipe, or the first (in case this particular easy
2430 isn't already) */
2431 if(entry->easy == data) {
2432 if(Curl_sendpipe_head(data, easy_conn))
2433 entry->easy = easy_conn->send_pipe.head->next->ptr;
2434 else
2435 entry->easy = easy_conn->send_pipe.head->ptr;
2436 }
2437 }
2438 /* Don't worry about overwriting recv_pipe head with send_pipe_head,
2439 when action will be asked on the socket (see multi_socket()), the
2440 head of the correct pipe will be taken according to the
2441 action. */
2442 }
2443
2444 if(remove_sock_from_hash) {
2445 /* in this case 'entry' is always non-NULL */
2446 if(multi->socket_cb)
2447 multi->socket_cb(data,
2448 s,
2449 CURL_POLL_REMOVE,
2450 multi->socket_userp,
2451 entry->socketp);
2452 sh_delentry(&multi->sockhash, s);
2453 }
2454 } /* if sockhash entry existed */
2455 } /* for loop over numsocks */
2456
2457 memcpy(data->sockets, socks, num*sizeof(curl_socket_t));
2458 data->numsocks = num;
2459}

2491/*
2492 * add_next_timeout()
2493 *
2494 * Each Curl_easy has a list of timeouts. The add_next_timeout() is called
2495 * when it has just been removed from the splay tree because the timeout has
2496 * expired. This function is then to advance in the list to pick the next
2497 * timeout to use (skip the already expired ones) and add this node back to
2498 * the splay tree again.
2499 *
2500 * The splay tree only has each sessionhandle as a single node and the nearest
2501 * timeout is used to sort it on.
2502 */
2503static CURLMcode add_next_timeout(struct curltime now,
2504 struct Curl_multi *multi,
2505 struct Curl_easy *d)
2506{
2507 struct curltime *tv = &d->state.expiretime;
2508 struct curl_llist *list = &d->state.timeoutlist;
2509 struct curl_llist_element *e;
2510 struct time_node *node = NULL;
2511
2512 /* move over the timeout list for this specific handle and remove all
2513 timeouts that are now passed tense and store the next pending
2514 timeout in *tv */
2515 for(e = list->head; e;) {
2516 struct curl_llist_element *n = e->next;
2517 time_t diff;
2518 node = (struct time_node *)e->ptr;
2519 diff = curlx_tvdiff(node->time, now);
2520 if(diff <= 0)
2521 /* remove outdated entry */
2522 Curl_llist_remove(list, e, NULL);
2523 else
2524 /* the list is sorted so get out on the first mismatch */
2525 break;
2526 e = n;
2527 }
2528 e = list->head;
2529 if(!e) {
2530 /* clear the expire times within the handles that we remove from the
2531 splay tree */
2532 tv->tv_sec = 0;
2533 tv->tv_usec = 0;
2534 }
2535 else {
2536 /* copy the first entry to 'tv' */
2537 memcpy(tv, &node->time, sizeof(*tv));
2538
2539 /* Insert this node again into the splay. Keep the timer in the list in
2540 case we need to recompute future timers. */
2541 multi->timetree = Curl_splayinsert(*tv, multi->timetree,
2542 &d->state.timenode);
2543 }
2544 return CURLM_OK;
2545}

2547static CURLMcode multi_socket(struct Curl_multi *multi,
2548 bool checkall,
2549 curl_socket_t s,
2550 int ev_bitmask,
2551 int *running_handles)
2552{
2553 CURLMcode result = CURLM_OK;
2554 struct Curl_easy *data = NULL;
2555 struct Curl_tree *t;
2556 struct curltime now = Curl_tvnow();
2557
2558 if(checkall) {
2559 /* *perform() deals with running_handles on its own */
2560 result = curl_multi_perform(multi, running_handles);
2561
2562 /* walk through each easy handle and do the socket state change magic
2563 and callbacks */
2564 if(result != CURLM_BAD_HANDLE) {
2565 data = multi->easyp;
2566 while(data) {
2567 singlesocket(multi, data);
2568 data = data->next;
2569 }
2570 }
2571
2572 /* or should we fall-through and do the timer-based stuff? */
2573 return result;
2574 }
2575 if(s != CURL_SOCKET_TIMEOUT) {
2576
2577 struct Curl_sh_entry *entry = sh_getentry(&multi->sockhash, s);
2578
2579 if(!entry)
2580 /* Unmatched socket, we can't act on it but we ignore this fact. In
2581 real-world tests it has been proved that libevent can in fact give
2582 the application actions even though the socket was just previously
2583 asked to get removed, so thus we better survive stray socket actions
2584 and just move on. */
2585 ;
2586 else {
2587 SIGPIPE_VARIABLE(pipe_st);
2588
2589 data = entry->easy;
2590
2591 if(data->magic != CURLEASY_MAGIC_NUMBER)
2592 /* bad bad bad bad bad bad bad */
2593 return CURLM_INTERNAL_ERROR;
2594
2595 /* If the pipeline is enabled, take the handle which is in the head of
2596 the pipeline. If we should write into the socket, take the send_pipe
2597 head. If we should read from the socket, take the recv_pipe head. */
2598 if(data->easy_conn) {
2599 if((ev_bitmask & CURL_POLL_OUT) &&
2600 data->easy_conn->send_pipe.head)
2601 data = data->easy_conn->send_pipe.head->ptr;
2602 else if((ev_bitmask & CURL_POLL_IN) &&
2603 data->easy_conn->recv_pipe.head)
2604 data = data->easy_conn->recv_pipe.head->ptr;
2605 }
2606
2607 if(data->easy_conn &&
2608 !(data->easy_conn->handler->flags & PROTOPT_DIRLOCK))
2609 /* set socket event bitmask if they're not locked */
2610 data->easy_conn->cselect_bits = ev_bitmask;
2611
2612 sigpipe_ignore(data, &pipe_st);
2613 result = multi_runsingle(multi, now, data);
2614 sigpipe_restore(&pipe_st);
2615
2616 if(data->easy_conn &&
2617 !(data->easy_conn->handler->flags & PROTOPT_DIRLOCK))
2618 /* clear the bitmask only if not locked */
2619 data->easy_conn->cselect_bits = 0;
2620
2621 if(CURLM_OK >= result)
2622 /* get the socket(s) and check if the state has been changed since
2623 last */
2624 singlesocket(multi, data);
2625
2626 /* Now we fall-through and do the timer-based stuff, since we don't want
2627 to force the user to have to deal with timeouts as long as at least
2628 one connection in fact has traffic. */
2629
2630 data = NULL; /* set data to NULL again to avoid calling
2631 multi_runsingle() in case there's no need to */
2632 now = Curl_tvnow(); /* get a newer time since the multi_runsingle() loop
2633 may have taken some time */
2634 }
2635 }
2636 else {
2637 /* Asked to run due to time-out. Clear the 'lastcall' variable to force
2638 update_timer() to trigger a callback to the app again even if the same
2639 timeout is still the one to run after this call. That handles the case
2640 when the application asks libcurl to run the timeout prematurely. */
2641 memset(&multi->timer_lastcall, 0, sizeof(multi->timer_lastcall));
2642 }
2643
2644 /*
2645 * The loop following here will go on as long as there are expire-times left
2646 * to process in the splay and 'data' will be re-assigned for every expired
2647 * handle we deal with.
2648 */
2649 do {
2650 /* the first loop lap 'data' can be NULL */
2651 if(data) {
2652 SIGPIPE_VARIABLE(pipe_st);
2653
2654 sigpipe_ignore(data, &pipe_st);
2655 result = multi_runsingle(multi, now, data);
2656 sigpipe_restore(&pipe_st);
2657
2658 if(CURLM_OK >= result)
2659 /* get the socket(s) and check if the state has been changed since
2660 last */
2661 singlesocket(multi, data);
2662 }
2663
2664 /* Check if there's one (more) expired timer to deal with! This function
2665 extracts a matching node if there is one */
2666
2667 multi->timetree = Curl_splaygetbest(now, multi->timetree, &t);
2668 if(t) {
2669 data = t->payload; /* assign this for next loop */
2670 (void)add_next_timeout(now, multi, t->payload);
2671 }
2672
2673 } while(t);
2674
2675 *running_handles = multi->num_alive;
2676 return result;
2677}

2779static CURLMcode multi_timeout(struct Curl_multi *multi,
2780 long *timeout_ms)
2781{
2782 static struct curltime tv_zero = {0, 0};
2783
2784 if(multi->timetree) {
2785 /* we have a tree of expire times */
2786 struct curltime now = Curl_tvnow();
2787
2788 /* splay the lowest to the bottom */
2789 multi->timetree = Curl_splay(tv_zero, multi->timetree);
2790
2791 if(Curl_splaycomparekeys(multi->timetree->key, now) > 0) {
2792 /* some time left before expiration */
2793 *timeout_ms = (long)curlx_tvdiff(multi->timetree->key, now);
2794 if(!*timeout_ms)
2795 /*
2796 * Since we only provide millisecond resolution on the returned value
2797 * and the diff might be less than one millisecond here, we don't
2798 * return zero as that may cause short bursts of busyloops on fast
2799 * processors while the diff is still present but less than one
2800 * millisecond! instead we return 1 until the time is ripe.
2801 */
2802 *timeout_ms = 1;
2803 }
2804 else
2805 /* 0 means immediately */
2806 *timeout_ms = 0;
2807 }
2808 else
2809 *timeout_ms = -1;
2810
2811 return CURLM_OK;
2812}

2824/*
2825 * Tell the application it should update its timers, if it subscribes to the
2826 * update timer callback.
2827 */
2828static int update_timer(struct Curl_multi *multi)
2829{
2830 long timeout_ms;
2831
2832 if(!multi->timer_cb)
2833 return 0;
2834 if(multi_timeout(multi, &timeout_ms)) {
2835 return -1;
2836 }
2837 if(timeout_ms < 0) {
2838 static const struct curltime none = {0, 0};
2839 if(Curl_splaycomparekeys(none, multi->timer_lastcall)) {
2840 multi->timer_lastcall = none;
2841 /* there's no timeout now but there was one previously, tell the app to
2842 disable it */
2843 return multi->timer_cb(multi, -1, multi->timer_userp);
2844 }
2845 return 0;
2846 }
2847
2848 /* When multi_timeout() is done, multi->timetree points to the node with the
2849 * timeout we got the (relative) time-out time for. We can thus easily check
2850 * if this is the same (fixed) time as we got in a previous call and then
2851 * avoid calling the callback again. */
2852 if(Curl_splaycomparekeys(multi->timetree->key, multi->timer_lastcall) == 0)
2853 return 0;
2854
2855 multi->timer_lastcall = multi->timetree->key;
2856
2857 return multi->timer_cb(multi, timeout_ms, multi->timer_userp);
2858}

2880/*
2881 * multi_addtimeout()
2882 *
2883 * Add a timestamp to the list of timeouts. Keep the list sorted so that head
2884 * of list is always the timeout nearest in time.
2885 *
2886 */
2887static CURLMcode
2888multi_addtimeout(struct Curl_easy *data,
2889 struct curltime *stamp,
2890 expire_id eid)
2891{
2892 struct curl_llist_element *e;
2893 struct time_node *node;
2894 struct curl_llist_element *prev = NULL;
2895 size_t n;
2896 struct curl_llist *timeoutlist = &data->state.timeoutlist;
2897
2898 node = &data->state.expires[eid];
2899
2900 /* copy the timestamp and id */
2901 memcpy(&node->time, stamp, sizeof(*stamp));
2902 node->eid = eid; /* also marks it as in use */
2903
2904 n = Curl_llist_count(timeoutlist);
2905 if(n) {
2906 /* find the correct spot in the list */
2907 for(e = timeoutlist->head; e; e = e->next) {
2908 struct time_node *check = (struct time_node *)e->ptr;
2909 time_t diff = curlx_tvdiff(check->time, node->time);
2910 if(diff > 0)
2911 break;
2912 prev = e;
2913 }
2914
2915 }
2916 /* else
2917 this is the first timeout on the list */
2918
2919 Curl_llist_insert_next(timeoutlist, prev, node, &node->list);
2920 return CURLM_OK;
2921}

2923/*
2924 * Curl_expire()
2925 *
2926 * given a number of milliseconds from now to use to set the 'act before
2927 * this'-time for the transfer, to be extracted by curl_multi_timeout()
2928 *
2929 * The timeout will be added to a queue of timeouts if it defines a moment in
2930 * time that is later than the current head of queue.
2931 *
2932 * Expire replaces a former timeout using the same id if already set.
2933 */
2934void Curl_expire(struct Curl_easy *data, time_t milli, expire_id id)
2935{
2936 struct Curl_multi *multi = data->multi;
2937 struct curltime *nowp = &data->state.expiretime;
2938 int rc;
2939 struct curltime set;
2940
2941 /* this is only interesting while there is still an associated multi struct
2942 remaining! */
2943 if(!multi)
2944 return;
2945
2946 DEBUGASSERT(id < EXPIRE_LAST);
2947
2948 set = Curl_tvnow();
2949 set.tv_sec += milli/1000;
2950 set.tv_usec += (unsigned int)(milli%1000)*1000;
2951
2952 if(set.tv_usec >= 1000000) {
2953 set.tv_sec++;
2954 set.tv_usec -= 1000000;
2955 }
2956
2957 /* Remove any timer with the same id just in case. */
2958 multi_deltimeout(data, id);
2959
2960 /* Add it to the timer list. It must stay in the list until it has expired
2961 in case we need to recompute the minimum timer later. */
2962 multi_addtimeout(data, &set, id);
2963
2964 if(nowp->tv_sec || nowp->tv_usec) {
2965 /* This means that the struct is added as a node in the splay tree.
2966 Compare if the new time is earlier, and only remove-old/add-new if it
2967 is. */
2968 time_t diff = curlx_tvdiff(set, *nowp);
2969
2970 if(diff > 0) {
2971 /* The current splay tree entry is sooner than this new expiry time.
2972 We don't need to update our splay tree entry. */
2973 return;
2974 }
2975
2976 /* Since this is an updated time, we must remove the previous entry from
2977 the splay tree first and then re-add the new value */
2978 rc = Curl_splayremovebyaddr(multi->timetree,
2979 &data->state.timenode,
2980 &multi->timetree);
2981 if(rc)
2982 infof(data, "Internal error removing splay node = %d\n", rc);
2983 }
2984
2985 /* Indicate that we are in the splay tree and insert the new timer expiry
2986 value since it is our local minimum. */
2987 *nowp = set;
2988 data->state.timenode.payload = data;
2989 multi->timetree = Curl_splayinsert(*nowp, multi->timetree,
2990 &data->state.timenode);
2991}
