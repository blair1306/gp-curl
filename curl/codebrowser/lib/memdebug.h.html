80/* Set this symbol on the command-line, recompile all lib-sources */
81#undef strdup
82#define strdup(ptr) curl_dostrdup(ptr, __LINE__, __FILE__)
83#define malloc(size) curl_domalloc(size, __LINE__, __FILE__)
84#define calloc(nbelem,size) curl_docalloc(nbelem, size, __LINE__, __FILE__)
85#define realloc(ptr,size) curl_dorealloc(ptr, size, __LINE__, __FILE__)
86#define free(ptr) curl_dofree(ptr, __LINE__, __FILE__)
87
88#ifdef WIN32
89# ifdef UNICODE
90# undef wcsdup
91# define wcsdup(ptr) curl_dowcsdup(ptr, __LINE__, __FILE__)
92# undef _wcsdup
93# define _wcsdup(ptr) curl_dowcsdup(ptr, __LINE__, __FILE__)
94# undef _tcsdup
95# define _tcsdup(ptr) curl_dowcsdup(ptr, __LINE__, __FILE__)
96# else
97# undef _tcsdup
98# define _tcsdup(ptr) curl_dostrdup(ptr, __LINE__, __FILE__)
99# endif
100#endif

164/*
165 * Curl_safefree defined as a macro to allow MemoryTracking feature
166 * to log free() calls at same location where Curl_safefree is used.
167 * This macro also assigns NULL to given pointer when free'd.
168 */
169
170#define Curl_safefree(ptr) \
171 do { free((ptr)); (ptr) = NULL;} WHILE_FALSE
