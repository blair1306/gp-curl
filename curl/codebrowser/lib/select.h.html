99/* Winsock and TPF sockets are not in range [0..FD_SETSIZE-1], which
100 unfortunately makes it impossible for us to easily check if they're valid
101*/
102#if defined(USE_WINSOCK) || defined(TPF)
103#define VALID_SOCK(x) 1
104#define VERIFY_SOCK(x) Curl_nop_stmt
105#else
106#define VALID_SOCK(s) (((s) >= 0) && ((s) < FD_SETSIZE))
107#define VERIFY_SOCK(x) do { \
108 if(!VALID_SOCK(x)) { \
109 SET_SOCKERRNO(EINVAL); \
110 return -1; \
111 } \
112} WHILE_FALSE
113#endif
