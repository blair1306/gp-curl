30 then it must be initialized to be used by NTLM. */

32 !defined(USE_GNUTLS_NETTLE) && \
33 !defined(USE_GNUTLS) && \
34 defined(USE_NSS)

44/* Define USE_NTRESPONSES in order to make the type-3 message include
45 * the NT response message. */
46#if !defined(USE_OPENSSL) || !defined(OPENSSL_NO_MD4)
47#define USE_NTRESPONSES
48#endif
49
50/* Define USE_NTLM2SESSION in order to make the type-3 message include the
51 NTLM2Session response message, requires USE_NTRESPONSES defined to 1 and a
52 Crypto engine that we have curl_ssl_md5sum() for. */
53#if defined(USE_NTRESPONSES) && !defined(USE_WIN32_CRYPTO)
54#define USE_NTLM2SESSION
55#endif
56
57/* Define USE_NTLM_V2 in order to allow the type-3 message to include the
58 LMv2 and NTLMv2 response messages, requires USE_NTRESPONSES defined to 1
59 and support for 64-bit integers. */
60#if defined(USE_NTRESPONSES) && (CURL_SIZEOF_CURL_OFF_T > 4)
61#define USE_NTLM_V2
62#endif
