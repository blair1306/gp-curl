47/****************************************************************************
48 * RTSP unique setup
49 ***************************************************************************/
50struct RTSP {
51 /*
52 * http_wrapper MUST be the first element of this structure for the wrap
53 * logic to work. In this way, we get a cheap polymorphism because
54 * &(data->state.proto.rtsp) == &(data->state.proto.http) per the C spec
55 *
56 * HTTP functions can safely treat this as an HTTP struct, but RTSP aware
57 * functions can also index into the later elements.
58 */
59 struct HTTP http_wrapper; /*wrap HTTP to do the heavy lifting */
60
61 long CSeq_sent; /* CSeq of this request */
62 long CSeq_recv; /* CSeq received */
63};
