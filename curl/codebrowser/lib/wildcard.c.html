33CURLcode Curl_wildcard_init(struct WildcardData *wc)
34{
35 Curl_llist_init(&wc->filelist, Curl_fileinfo_dtor);
36 wc->state = CURLWC_INIT;
37
38 return CURLE_OK;
39}
40
41void Curl_wildcard_dtor(struct WildcardData *wc)
42{
43 if(!wc)
44 return;
45
46 if(wc->tmp_dtor) {
47 wc->tmp_dtor(wc->tmp);
48 wc->tmp_dtor = ZERO_NULL;
49 wc->tmp = NULL;
50 }
51 DEBUGASSERT(wc->tmp == NULL);
52
53 Curl_llist_destroy(&wc->filelist, NULL);
54
55
56 free(wc->path);
57 wc->path = NULL;
58 free(wc->pattern);
59 wc->pattern = NULL;
60
61 wc->customptr = NULL;
62 wc->state = CURLWC_INIT;
63}
