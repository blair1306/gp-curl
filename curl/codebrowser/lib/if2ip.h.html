49/* Nedelcho Stanev's work-around for SFU 3.0 */
50struct ifreq {
51#define IFNAMSIZ 16
52#define IFHWADDRLEN 6
53 union {
54 char ifrn_name[IFNAMSIZ]; /* if name, e.g. "en0" */
55 } ifr_ifrn;
56
57 union {
58 struct sockaddr ifru_addr;
59 struct sockaddr ifru_broadaddr;
60 struct sockaddr ifru_netmask;
61 struct sockaddr ifru_hwaddr;
62 short ifru_flags;
63 int ifru_metric;
64 int ifru_mtu;
65 } ifr_ifru;
66};
67
68/* This define was added by Daniel to avoid an extra #ifdef INTERIX in the
69 C code. */
70
71#define ifr_name ifr_ifrn.ifrn_name /* interface name */
72#define ifr_addr ifr_ifru.ifru_addr /* address */
73#define ifr_broadaddr ifr_ifru.ifru_broadaddr /* broadcast address */
74#define ifr_netmask ifr_ifru.ifru_netmask /* interface net mask */
75#define ifr_flags ifr_ifru.ifru_flags /* flags */
76#define ifr_hwaddr ifr_ifru.ifru_hwaddr /* MAC address */
77#define ifr_metric ifr_ifru.ifru_metric /* metric */
78#define ifr_mtu ifr_ifru.ifru_mtu /* mtu */
79
80#define SIOCGIFADDR _IOW('s', 102, struct ifreq) /* Get if addr */
