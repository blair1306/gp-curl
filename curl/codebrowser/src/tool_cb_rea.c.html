33/*
34** callback for CURLOPT_READFUNCTION
35*/
36
37size_t tool_read_cb(void *buffer, size_t sz, size_t nmemb, void *userdata)
38{
39 ssize_t rc;
40 struct InStruct *in = userdata;
41
42 rc = read(in->fd, buffer, sz*nmemb);
43 if(rc < 0) {
44 if(errno == EAGAIN) {
45 errno = 0;
46 in->config->readbusy = TRUE;
47 return CURL_READFUNC_PAUSE;
48 }
49 /* since size_t is unsigned we can't return negative values fine */
50 rc = 0;
51 }
52 in->config->readbusy = FALSE;
53 return (size_t)rc;
54}
