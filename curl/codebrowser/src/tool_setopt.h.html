30#define SETOPT_CHECK(v) do { \
31 result = (v); \
32 if(result) \
33 goto show_error; \
34} WHILE_FALSE

38/* Associate symbolic names with option values */
39typedef struct {
40 const char *name;
41 long value;
42} NameValue;
43
44typedef struct {
45 const char *name;
46 unsigned long value;
47} NameValueUnsigned;

50extern const NameValue setopt_nv_CURL_SOCKS_PROXY[];

53extern const NameValue setopt_nv_CURL_TIMECOND[];
54extern const NameValue setopt_nv_CURLFTPSSL_CCC[];
55extern const NameValue setopt_nv_CURLUSESSL[];

57extern const NameValue setopt_nv_CURL_NETRC[];
58extern const NameValue setopt_nv_CURLPROTO[];
